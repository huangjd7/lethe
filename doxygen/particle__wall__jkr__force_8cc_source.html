10template <int dim, typename PropertiesIndex>
11ParticleWallJKRForce<dim, PropertiesIndex>::ParticleWallJKRForce(
12 const DEMSolverParameters<dim> &dem_parameters,
13 const std::vector<types::boundary_id> &boundary_index)
14 : ParticleWallContactForce<dim, PropertiesIndex>(dem_parameters)
15 , f_coefficient_epsd(dem_parameters.model_parameters.f_coefficient_epsd)
16
17{
18 // Wall properties
19 const double wall_youngs_modulus =
20 dem_parameters.lagrangian_physical_properties.youngs_modulus_wall;
21 const double wall_poisson_ratio =
22 dem_parameters.lagrangian_physical_properties.poisson_ratio_wall;
23 const double wall_restitution_coefficient =
24 dem_parameters.lagrangian_physical_properties.restitution_coefficient_wall;
25 const double wall_friction_coefficient =
26 dem_parameters.lagrangian_physical_properties.friction_coefficient_wall;
27 const double wall_rolling_friction_coefficient =
28 dem_parameters.lagrangian_physical_properties.rolling_friction_wall;
29 const double wall_rolling_viscous_damping =
30 dem_parameters.lagrangian_physical_properties.rolling_viscous_damping_wall;
31 const double wall_surface_energy =
32 dem_parameters.lagrangian_physical_properties.surface_energy_wall;
33
34 for (unsigned int i = 0;
35 i < dem_parameters.lagrangian_physical_properties.particle_type_number;
36 ++i)
37 {
38 // Particle properties
39 const double particle_youngs_modulus =
40 dem_parameters.lagrangian_physical_properties.youngs_modulus_particle
41 .at(i);
42 const double particle_poisson_ratio =
43 dem_parameters.lagrangian_physical_properties.poisson_ratio_particle.at(
44 i);
45 const double particle_restitution_coefficient =
46 dem_parameters.lagrangian_physical_properties
47 .restitution_coefficient_particle.at(i);
48 const double particle_friction_coefficient =
49 dem_parameters.lagrangian_physical_properties
50 .friction_coefficient_particle.at(i);
51 const double particle_rolling_friction_coefficient =
52 dem_parameters.lagrangian_physical_properties
53 .rolling_friction_coefficient_particle.at(i);
54 const double particle_rolling_viscous_damping_coefficient =
55 dem_parameters.lagrangian_physical_properties
56 .rolling_viscous_damping_coefficient_particle.at(i);
57 const double particle_surface_energy =
58 dem_parameters.lagrangian_physical_properties.surface_energy_particle
59 .at(i);
60
61 // Effective particle-wall properties.
62 this->effective_youngs_modulus[i] =
63 (particle_youngs_modulus * wall_youngs_modulus) /
64 (wall_youngs_modulus *
65 (1 - particle_poisson_ratio * particle_poisson_ratio) +
66 particle_youngs_modulus *
67 (1 - wall_poisson_ratio * wall_poisson_ratio) +
68 DBL_MIN);
69
70 this->effective_shear_modulus[i] =
71 (particle_youngs_modulus * wall_youngs_modulus) /
72 ((2 * wall_youngs_modulus * (2 - particle_poisson_ratio) *
73 (1 + particle_poisson_ratio)) +
74 (2 * particle_youngs_modulus * (2 - wall_poisson_ratio) *
75 (1 + wall_poisson_ratio)) +
76 DBL_MIN);
77
78 this->effective_coefficient_of_restitution[i] =
79 2 * particle_restitution_coefficient * wall_restitution_coefficient /
80 (particle_restitution_coefficient + wall_restitution_coefficient +
81 DBL_MIN);
82
83 this->effective_coefficient_of_friction[i] =
84 2 * particle_friction_coefficient * wall_friction_coefficient /
85 (particle_friction_coefficient + wall_friction_coefficient + DBL_MIN);
86
87 this->effective_coefficient_of_rolling_friction[i] =
88 2 * particle_rolling_friction_coefficient *
89 wall_rolling_friction_coefficient /
90 (particle_rolling_friction_coefficient +
91 wall_rolling_friction_coefficient + DBL_MIN);
92
93 this->effective_coefficient_of_rolling_viscous_damping[i] =
94 2 * particle_rolling_viscous_damping_coefficient *
95 wall_rolling_viscous_damping /
96 (particle_rolling_viscous_damping_coefficient +
97 wall_rolling_viscous_damping + DBL_MIN);
98
99 this->effective_surface_energy[i] =
100 particle_surface_energy + wall_surface_energy -
101 std::pow(std::sqrt(particle_surface_energy) -
102 std::sqrt(wall_surface_energy),
103 2);
104
105 const double log_coeff_restitution =
106 log(this->effective_coefficient_of_restitution[i]);
107 this->model_parameter_beta[i] =
108 log_coeff_restitution /
109 sqrt((log_coeff_restitution * log_coeff_restitution) + 9.8696);
110 }
111
112 if (dem_parameters.model_parameters.rolling_resistance_method ==
113 Parameters::Lagrangian::RollingResistanceMethod::no_resistance)
114 {
115 calculate_rolling_resistance_torque =
116 &ParticleWallJKRForce<dim, PropertiesIndex>::no_resistance;
117 }
118 else if (dem_parameters.model_parameters.rolling_resistance_method ==
119 Parameters::Lagrangian::RollingResistanceMethod::constant_resistance)
120 {
121 calculate_rolling_resistance_torque =
122 &ParticleWallJKRForce<dim, PropertiesIndex>::constant_resistance;
123 }
124 else if (dem_parameters.model_parameters.rolling_resistance_method ==
125 Parameters::Lagrangian::RollingResistanceMethod::viscous_resistance)
126 {
127 calculate_rolling_resistance_torque =
128 &ParticleWallJKRForce<dim, PropertiesIndex>::viscous_resistance;
129 }
130 else if (dem_parameters.model_parameters.rolling_resistance_method ==
131 Parameters::Lagrangian::RollingResistanceMethod::epsd_resistance)
132 {
133 calculate_rolling_resistance_torque =
134 &ParticleWallJKRForce<dim, PropertiesIndex>::epsd_resistance;
135 }
136
137
138 this->calculate_force_torque_on_boundary =
139 dem_parameters.forces_torques.calculate_force_torque;
140 this->center_mass_container = dem_parameters.forces_torques.point_center_mass;
141 this->boundary_index = boundary_index;
142 this->force_on_walls = this->initialize();
143 this->torque_on_walls = this->initialize();
144}

146template <int dim, typename PropertiesIndex>
147
148void
149ParticleWallJKRForce<dim, PropertiesIndex>::
150 calculate_particle_wall_contact_force(
151 typename DEM::dem_data_structures<dim>::particle_wall_in_contact
152 &particle_wall_pairs_in_contact,
153 const double dt,
154 std::vector<Tensor<1, 3>> &torque,
155 std::vector<Tensor<1, 3>> &force)
156{
157 ParticleWallContactForce<dim, PropertiesIndex>::force_on_walls =
158 ParticleWallContactForce<dim, PropertiesIndex>::initialize();
159 ParticleWallContactForce<dim, PropertiesIndex>::torque_on_walls =
160 ParticleWallContactForce<dim, PropertiesIndex>::initialize();
161
162 // Looping over particle_wall_pairs_in_contact, which means looping over all
163 // the active particles with iterator particle_wall_pairs_in_contact_iterator
164 for (auto &&pairs_in_contact_content :
165 particle_wall_pairs_in_contact | boost::adaptors::map_values)
166 {
167 // Now an iterator (particle_wall_contact_information_iterator) on each
168 // element of the particle_wall_pairs_in_contact vector is defined. This
169 // iterator iterates over a map which contains the required information
170 // for calculation of the contact force for each particle
171 for (auto &&contact_information :
172 pairs_in_contact_content | boost::adaptors::map_values)
173 {
174 // Defining the total force of contact, properties of particle as
175 // local parameters
176 auto particle = contact_information.particle;
177 auto particle_properties = particle->get_properties();
178
179 auto normal_vector = contact_information.normal_vector;
180 auto point_on_boundary = contact_information.point_on_boundary;
181
182
183 Point<3> particle_location_3d = [&] {
184 if constexpr (dim == 3)
185 {
186 return particle->get_location();
187 }
188 if constexpr (dim == 2)
189 {
190 return (point_nd_to_3d(particle->get_location()));
191 }
192 }();
193
194 // A vector (point_to_particle_vector) is defined which connects the
195 // center of particle to the point_on_boundary. This vector will then
196 // be projected on the normal vector of the boundary to obtain the
197 // particle-wall distance
198 Tensor<1, 3> point_to_particle_vector =
199 particle_location_3d - point_on_boundary;
200
201 // Finding the projected vector on the normal vector of the boundary.
202 // Here we have used the private function find_projection. Using this
203 // projected vector, the particle-wall distance is calculated
204 Tensor<1, 3> projected_vector =
205 this->find_projection(point_to_particle_vector, normal_vector);
206
207 double normal_overlap =
208 ((particle_properties[PropertiesIndex::dp]) * 0.5) -
209 (projected_vector.norm());
210
211 if (normal_overlap > 0.)
212 {
213 contact_information.normal_overlap = normal_overlap;
214
215 this->update_contact_information(contact_information,
216 particle_location_3d,
217 particle_properties,
218 dt);
219
220 // This tuple (forces and torques) contains four elements which
221 // are: 1, normal force, 2, tangential force, 3, tangential torque
222 // and 4, rolling resistance torque, respectively
223 std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
224 forces_and_torques =
225 this->calculate_jkr_contact_force_and_torque(
226 dt, contact_information, particle_properties);
227
228 // Get particle's torque and force
229 types::particle_index particle_id = particle->get_local_index();
230
231 Tensor<1, 3> &particle_torque = torque[particle_id];
232 Tensor<1, 3> &particle_force = force[particle_id];
233
234 // Apply the calculated forces and torques on the particle
235 this->apply_force_and_torque(forces_and_torques,
236 particle_torque,
237 particle_force,
238 point_on_boundary,
239 contact_information.boundary_id);
240 }
241 else
242 {
243 contact_information.normal_overlap = 0.;
244 contact_information.tangential_overlap.clear();
245 contact_information.rolling_resistance_spring_torque.clear();
246 }
247 }
248 }
249}

252template <int dim, typename PropertiesIndex>
253void
254ParticleWallJKRForce<dim, PropertiesIndex>::
255 calculate_particle_floating_wall_contact_force(
256 typename DEM::dem_data_structures<dim>::particle_floating_mesh_in_contact
257 &particle_floating_mesh_in_contact,
258 const double dt,
259 std::vector<Tensor<1, 3>> &torque,
260 std::vector<Tensor<1, 3>> &force,
261 const std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> &solids)
262{
263 std::vector<Particles::ParticleIterator<dim>> particle_locations;
264 std::vector<Point<dim>> triangle(this->vertices_per_triangle);
265
266 for (unsigned int solid_counter = 0; solid_counter < solids.size();
267 ++solid_counter)
268 {
269 // Get translational and rotational velocities and
270 // center of rotation
271 Tensor<1, 3> translational_velocity =
272 solids[solid_counter]->get_translational_velocity();
273 Tensor<1, 3> angular_velocity =
274 solids[solid_counter]->get_angular_velocity();
275 Point<3> center_of_rotation =
276 solids[solid_counter]->get_center_of_rotation();
277
278 auto &particle_floating_mesh_contact_pair =
279 particle_floating_mesh_in_contact[solid_counter];
280
281 for (auto &[cut_cell, map_info] : particle_floating_mesh_contact_pair)
282 {
283 if (!map_info.empty())
284 {
285 // Clear the particle locations vector for the new cut cell
286 particle_locations.clear();
287 const unsigned int n_particles = map_info.size();
288
289 // Gather all the particles locations in a vector
290 for (auto &&contact_info : map_info | boost::adaptors::map_values)
291 {
292 particle_locations.push_back(contact_info.particle);
293 }
294
295 // Build triangle vector
296 for (unsigned int vertex = 0;
297 vertex < this->vertices_per_triangle;
298 ++vertex)
299 {
300 // Find vertex-floating wall distance
301 triangle[vertex] = cut_cell->vertex(vertex);
302 }
303
304 // Call find_particle_triangle_projection to get the
305 // distance and projection of particles on the triangle
306 // (floating mesh cell)
307 auto particle_triangle_information = LetheGridTools::
308 find_particle_triangle_projection<dim, PropertiesIndex>(
309 triangle, particle_locations, n_particles);
310
311 const std::vector<bool> pass_distance_check =
312 std::get<0>(particle_triangle_information);
313 const std::vector<Point<3>> projection_points =
314 std::get<1>(particle_triangle_information);
315 const std::vector<Tensor<1, 3>> normal_vectors =
316 std::get<2>(particle_triangle_information);
317
318 unsigned int particle_counter = 0;
319
320 for (auto &&contact_info : map_info | boost::adaptors::map_values)
321 {
322 // If particle passes the distance check
323 if (pass_distance_check[particle_counter])
324 {
325 // Define the total force of contact, properties of
326 // particle as local parameters
327 auto &particle = contact_info.particle;
328 auto particle_properties = particle->get_properties();
329
330 const Point<3> &projection_point =
331 projection_points[particle_counter];
332
333 Point<3> particle_location_3d;
334
335 if constexpr (dim == 3)
336 particle_location_3d = particle->get_location();
337
338 if constexpr (dim == 2)
339 particle_location_3d =
340 point_nd_to_3d(particle->get_location());
341
342 const double particle_triangle_distance =
343 particle_location_3d.distance(projection_point);
344
345 // Find normal overlap
346 double normal_overlap =
347 ((particle_properties[PropertiesIndex::dp]) * 0.5) -
348 particle_triangle_distance;
349
350 if (normal_overlap > 0)
351 {
352 contact_info.normal_overlap = normal_overlap;
353
354 contact_info.normal_vector =
355 normal_vectors[particle_counter];
356
357 contact_info.point_on_boundary = projection_point;
358
359 contact_info.boundary_id = solid_counter;
360
361 this
362 ->update_particle_floating_wall_contact_information(
363 contact_info,
364 particle_properties,
365 dt,
366 translational_velocity,
367 angular_velocity,
368 center_of_rotation.distance(
369 particle_location_3d));
370
371 // This tuple (forces and torques) contains four
372 // elements which are: 1, normal force, 2, tangential
373 // force, 3, tangential torque and 4, rolling
374 // resistance torque, respectively
375 std::tuple<Tensor<1, 3>,
376 Tensor<1, 3>,
377 Tensor<1, 3>,
378 Tensor<1, 3>>
379 forces_and_torques =
380 this->calculate_jkr_contact_force_and_torque(
381 dt, contact_info, particle_properties);
382
383 // Get particle's torque and force
384 types::particle_index particle_id =
385 particle->get_local_index();
386
387 Tensor<1, 3> &particle_torque = torque[particle_id];
388 Tensor<1, 3> &particle_force = force[particle_id];
389
390 // Apply the calculated forces and torques on the
391 // particle
392 this->apply_force_and_torque(
393 forces_and_torques,
394 particle_torque,
395 particle_force,
396 projection_point,
397 contact_info.boundary_id);
398 }
399 else
400 {
401 contact_info.normal_overlap = 0;
402 contact_info.tangential_overlap.clear();
403 contact_info.rolling_resistance_spring_torque.clear();
404 }
405 }
406 particle_counter++;
407 }
408 }
409 }
410 }

413// Calculates JKR contact force and torques
414template <int dim, typename PropertiesIndex>
415std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
416ParticleWallJKRForce<dim, PropertiesIndex>::
417 calculate_jkr_contact_force_and_torque(
418 const double dt,
419 particle_wall_contact_info<dim> &contact_info,
420 const ArrayView<const double> &particle_properties)
421{
422 // i is the particle, j is the wall.
423 // we need to put a minus sign infront of the normal_vector to respect the
424 // convention (i -> j)
425 Tensor<1, 3> normal_vector = -contact_info.normal_vector;
426 const unsigned int particle_type = particle_properties[PropertiesIndex::type];
427
428 const double effective_radius =
429 0.5 * particle_properties[PropertiesIndex::dp];
430
431 // Calculation of model parameters (beta, sn and st). These values
432 // are used to consider non-linear relation of the contact force to
433 // the normal overlap
434 double radius_times_overlap_sqrt =
435 sqrt(particle_properties[PropertiesIndex::dp] * 0.5 *
436 contact_info.normal_overlap);
437 double model_parameter_sn = 2 *
438 this->effective_youngs_modulus[particle_type] *
439 radius_times_overlap_sqrt;
440
441 double model_parameter_st = 8 * this->effective_shear_modulus[particle_type] *
442 radius_times_overlap_sqrt;
443
444 // Calculation of the contact patch radius (a) using the analytical solution
445 // describe in the theory guide.
446 const double c0 =
447 Utilities::fixed_power<2>(effective_radius * contact_info.normal_overlap);
448 const double c1 = -2. * Utilities::fixed_power<2>(effective_radius) * M_PI *
449 this->effective_surface_energy[particle_type] /
450 this->effective_youngs_modulus[particle_type];
451 const double c2 = -2. * contact_info.normal_overlap * effective_radius;
452 const double P = -Utilities::fixed_power<2>(c2) / 12. - c0;
453 const double Q = -Utilities::fixed_power<3>(c2) / 108. + c0 * c2 / 3. -
454 Utilities::fixed_power<2>(c1) * 0.125;
455 const double root1 = std::max(0.,
456 (0.25 * Utilities::fixed_power<2>(Q)) +
457 (Utilities::fixed_power<3>(P) / 27.));
458 const double U = std::cbrt(-0.5 * Q + std::sqrt(root1));
459 const double s = -c2 * (5. / 6.) + U - P / (3. * U);
460 const double w = std::sqrt(std::max(1e-16, c2 + 2. * s));
461 const double lambda = 0.5 * c1 / w;
462 const double root2 = std::max(1e-16, w * w - 4. * (c2 + s + lambda));
463 const double a = 0.5 * (w + std::sqrt(root2));
464
465 // Calculation of normal damping and tangential spring and dashpot constants
466 // using particle and wall properties.
467 // There is no minus sign here since model_parameter_beta is negative or
468 // equal to zero.
469 const double normal_damping_constant =
470 1.8257 * this->model_parameter_beta[particle_type] * // 2. * sqrt(5./6.)
471 sqrt(model_parameter_sn * particle_properties[PropertiesIndex::mass]);
472
473 // Tangential spring constant is set as a negative just like in the other
474 // particle-wall models. This must be taken into account for the square root
475 // in the tangential_damping_calculation.
476 double tangential_spring_constant =
477 -8. * this->effective_shear_modulus[particle_type] *
478 radius_times_overlap_sqrt;
479
480 // There is no minus sign here since model_parameter_beta is negative or
481 // equal to zero.
482 const double tangential_damping_constant =
483 normal_damping_constant *
484 sqrt(model_parameter_st / (model_parameter_sn + DBL_MIN));
485
486 // Calculation of the normal force coefficient (F_n_JKR)
487 const double normal_force_norm =
488 4. * this->effective_youngs_modulus[particle_type] *
489 Utilities::fixed_power<3>(a) / (3. * effective_radius) -
490 std::sqrt(8 * M_PI * this->effective_surface_energy[particle_type] *
491 this->effective_youngs_modulus[particle_type] *
492 Utilities::fixed_power<3>(a)) +
493 normal_damping_constant * contact_info.normal_relative_velocity;
494
495 // Calculation of normal force using the normal_force_norm and the
496 // normal vector.
497 Tensor<1, 3> normal_force = normal_force_norm * normal_vector;
498
499 // Calculation of tangential forces.
500 Tensor<1, 3> damping_tangential_force =
501 tangential_damping_constant * contact_info.tangential_relative_velocity;
502 Tensor<1, 3> tangential_force =
503 tangential_spring_constant * contact_info.tangential_overlap +
504 damping_tangential_force;
505 double tangential_force_norm = tangential_force.norm();
506
507 // JKR theory says that the coulomb threshold must be modified with the
508 // pull-out force. (Thornton 1991)
509 const double modified_coulomb_threshold =
510 (normal_force_norm + 3. * M_PI *
511 this->effective_surface_energy[particle_type] *
512 effective_radius) *
513 this->effective_coefficient_of_friction[particle_type];
514
515 // Check for gross sliding
516 if (tangential_force_norm > modified_coulomb_threshold)
517 {
518 // Gross sliding occurs and the tangential overlap and tangential
519 // force are limited to Coulomb's criterion
520 contact_info.tangential_overlap =
521 (modified_coulomb_threshold *
522 (tangential_force / (tangential_force_norm + DBL_MIN)) -
523 damping_tangential_force) /
524 (tangential_spring_constant + DBL_MIN);
525
526 tangential_force =
527 (tangential_spring_constant * contact_info.tangential_overlap) +
528 damping_tangential_force;
529 }
530
531 // Calculation torque caused by tangential force
532 // We add the minus sign here since the tangential_force is applied on the
533 // particle is in the opposite direction
534 Tensor<1, 3> tangential_torque =
535 cross_product_3d((0.5 * particle_properties[PropertiesIndex::dp] *
536 normal_vector),
537 -tangential_force);
538
539 // We need to compute the normal spring constant for case where we use the
540 // EPSD rolling resistance model.
541 double normal_spring_constant = 0.66665 * model_parameter_sn;
542
543 // Rolling resistance torque
544 Tensor<1, 3> rolling_resistance_torque =
545 (this->*calculate_rolling_resistance_torque)(
546 particle_properties,
547 this->effective_coefficient_of_rolling_friction[particle_type],
548 this->effective_coefficient_of_rolling_viscous_damping[particle_type],
549 normal_force.norm(),
550 dt,
551 normal_spring_constant,
552 contact_info.normal_vector,
553 contact_info.rolling_resistance_spring_torque);
554
555 return std::make_tuple(normal_force,
556 tangential_force,
557 tangential_torque,
558 rolling_resistance_torque);
559}
