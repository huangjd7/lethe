21template <int dim, int spacedim>
22void
23attach_grid_to_triangulation(
24 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
25 const Parameters::Mesh &mesh_parameters)
26
27{
28 // GMSH input
29 if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
30 {
31 if (mesh_parameters.simplex)
32 {
33 auto comm = triangulation.get_communicator();
34 std::string file_name = mesh_parameters.file_name;
35
36 auto construction_data = TriangulationDescription::Utilities::
37 create_description_from_triangulation_in_groups<dim, spacedim>(
38 [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
39 GridIn<dim, spacedim> grid_in;
40 grid_in.attach_triangulation(basetria);
41 std::ifstream input_file(file_name);
42
43 grid_in.read_msh(input_file);
44 },
45 [](dealii::Triangulation<dim, spacedim> &basetria,
46 const MPI_Comm comm,
47 const unsigned int /*group_size*/) {
48 GridTools::partition_triangulation(
49 Utilities::MPI::n_mpi_processes(comm), basetria);
50 },
51 comm,
52 Utilities::MPI::n_mpi_processes(comm) /* group size */,
53 dealii::Triangulation<dim, spacedim>::none);
54
55 triangulation.create_triangulation(construction_data);
56 GridTools::scale(mesh_parameters.scale, triangulation);
57 }
58 else
59 {
60 GridIn<dim, spacedim> grid_in;
61 grid_in.attach_triangulation(triangulation);
62 std::ifstream input_file(mesh_parameters.file_name);
63 grid_in.read_msh(input_file);
64 GridTools::scale(mesh_parameters.scale, triangulation);
65 }
66 }
67 // Dealii grids
68 else if (mesh_parameters.type == Parameters::Mesh::Type::dealii)
69 {
70 if (mesh_parameters.simplex)
71 {
72 Triangulation<dim, spacedim> temporary_quad_triangulation;
73 GridGenerator::generate_from_name_and_arguments(
74 temporary_quad_triangulation,
75 mesh_parameters.grid_type,
76 mesh_parameters.grid_arguments);
77
78 GridTools::scale(mesh_parameters.scale, temporary_quad_triangulation);
79
80 // initial refinement
81 const unsigned int initial_refinement =
82 mesh_parameters.initial_refinement;
83 temporary_quad_triangulation.refine_global(initial_refinement);
84
85 // flatten the triangulation
86 Triangulation<dim, spacedim> flat_temp_quad_triangulation;
87 GridGenerator::flatten_triangulation(temporary_quad_triangulation,
88 flat_temp_quad_triangulation);
89
90 Triangulation<dim, spacedim> temporary_tri_triangulation(
91 Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
92 GridGenerator::convert_hypercube_to_simplex_mesh(
93 flat_temp_quad_triangulation, temporary_tri_triangulation);
94
95 GridTools::partition_triangulation_zorder(
96 Utilities::MPI::n_mpi_processes(triangulation.get_communicator()),
97 temporary_tri_triangulation);
98 GridTools::partition_multigrid_levels(temporary_tri_triangulation);
99
100 // extract relevant information from distributed triangulation
101 auto construction_data = TriangulationDescription::Utilities::
102 create_description_from_triangulation(
103 temporary_tri_triangulation,
104 triangulation.get_communicator(),
105 TriangulationDescription::Settings::
106 construct_multigrid_hierarchy);
107 triangulation.create_triangulation(construction_data);
108 }
109 else
110 {
111 GridGenerator::generate_from_name_and_arguments(
112 triangulation,
113 mesh_parameters.grid_type,
114 mesh_parameters.grid_arguments);
115
116 GridTools::scale(mesh_parameters.scale, triangulation);
117 }
118 }
119 // Customizable cylinder mesh
120 else if (mesh_parameters.type == Parameters::Mesh::Type::cylinder)
121 {
122 if (mesh_parameters.simplex)
123 {
124 throw std::runtime_error(
125 "Unsupported mesh type - custom cylinder mesh with simplex is not supported. Use a dealii cylinder to use simplex mesh.");
126 }
127 else if (dim != 3)
128 {
129 throw std::runtime_error(
130 "Unsupported mesh type - custom cylinder mesh is only supported in 3d.");
131 }
132
133 if constexpr (dim == 3)
134 {
135 // Separate arguments of the string
136 std::vector<std::string> arguments;
137 std::stringstream s_stream(mesh_parameters.grid_arguments);
138 while (s_stream.good())
139 {
140 std::string substr;
141 getline(s_stream, substr, ':');
142 arguments.push_back(substr);
143 }
144
145 // Arguments declaration
146 unsigned int subdivisions;
147 double radius, half_height;
148 if (arguments.size() != 3)
149 {
150 throw std::runtime_error(
151 "Mandatory cylinder parameters are (x subdivisions: radius : half height)");
152 }
153 else
154 {
155 std::vector<double> arguments_double =
156 dealii::Utilities::string_to_double(arguments);
157 subdivisions = static_cast<int>(arguments_double[0]);
158 radius = arguments_double[1];
159 half_height = arguments_double[2];
160 }
161
162 if (mesh_parameters.grid_type == "classic")
163 {
164 // Create a subdivided cylinder from deal.ii
165 GridGenerator::subdivided_cylinder(triangulation,
166 subdivisions,
167 radius,
168 half_height);
169
170 GridTools::scale(mesh_parameters.scale, triangulation);
171 }
172 else
173 {
174 // Create a temporary 2d mesh
175 Triangulation<2, spacedim - 1> temporary_triangulation;
176
177 // Create a spherical manifold for 2d mesh
178 Point<2> center(0.0, 0.0);
179 const SphericalManifold<2, spacedim - 1> m0(center);
180
181 if (mesh_parameters.grid_type == "regularized" ||
182 mesh_parameters.grid_type == "squared")
183 {
184 // Create a square mesh
185 double real_radius = radius * std::sin(M_PI_4);
186 GridGenerator::hyper_cube(temporary_triangulation,
187 -real_radius,
188 real_radius,
189 true);
190
191 // Assign boundary 0 to perimeter as for cylinder
192 for (const auto &cell :
193 temporary_triangulation.active_cell_iterators())
194 {
195 if (cell->is_locally_owned())
196 {
197 // Looping through all the faces of the cell
198 for (const auto &face : cell->face_iterators())
199 {
200 // Check to see if the face is located at boundary
201 if (face->at_boundary())
202 {
203 face->set_boundary_id(0);
204 }
205 }
206 }
207 }
208 }
209 else if (mesh_parameters.grid_type == "balanced")
210 {
211 GridGenerator::hyper_ball_balanced(temporary_triangulation,
212 center,
213 radius);
214 }
215 else
216 {
217 throw std::runtime_error(
218 "Unknown grid type. Choices are <classic|balanced|squared|regularized>.");
219 }
220
221 temporary_triangulation.reset_all_manifolds();
222 temporary_triangulation.set_all_manifold_ids_on_boundary(0);
223 temporary_triangulation.set_manifold(0, m0);
224
225 if (mesh_parameters.grid_type == "regularized")
226 {
227 // Pre-refinement to reduce mesh size at corners before
228 // regularization
229 temporary_triangulation.refine_global(2);
230 GridTools::regularize_corner_cells(temporary_triangulation);
231
232 // Flatten the triangulation
233 Triangulation<2, spacedim - 1> flat_temporary_triangulation;
234 flat_temporary_triangulation.copy_triangulation(
235 temporary_triangulation);
236 temporary_triangulation.clear();
237 GridGenerator::flatten_triangulation(
238 flat_temporary_triangulation, temporary_triangulation);
239 }
240
241 // Extrude the 2d temporary mesh to 3d cylinder
242 GridGenerator::extrude_triangulation(temporary_triangulation,
243 subdivisions + 1,
244 2.0 * half_height,
245 triangulation,
246 true);
247
248 // Rotate mesh in x-axis and set the (0,0,0) at the barycenter
249 // to be comparable to dealii cylinder meshes
250 Tensor<1, spacedim> axis_vector({0.0, 1.0, 0.0});
251 GridTools::rotate(axis_vector, M_PI_2, triangulation);
252 Tensor<1, spacedim> shift_vector({-half_height, 0.0, 0.0});
253 GridTools::shift(shift_vector, triangulation);
254
255 // Force the manifold id to be zero in the case of the balanced
256 // cylinder
257 if (mesh_parameters.grid_type == "balanced")
258 triangulation.reset_manifold(1);
259
260 // Add a cylindrical manifold on the final unrefined mesh
261 const CylindricalManifold<3, spacedim> m1(0);
262 triangulation.set_manifold(0, m1);
263
264 GridTools::scale(mesh_parameters.scale, triangulation);
265 }
266 }
267 }
268
269 // Periodic Hills grid
270 else if (mesh_parameters.type == Parameters::Mesh::Type::periodic_hills &&
271 !mesh_parameters.simplex)
272 {
273 if (mesh_parameters.simplex)
274 {
275 throw std::runtime_error(
276 "Unsupported mesh type - periodic hills mesh with simplex is not supported");
277 }
278 else
279 {
280 PeriodicHillsGrid<dim, spacedim> grid(mesh_parameters.grid_arguments);
281 grid.make_grid(triangulation);
282
283 GridTools::scale(mesh_parameters.scale, triangulation);
284 }
285 }
286 else
287 throw std::runtime_error(
288 "Unsupported mesh type - mesh will not be created");
289}

292template <int dim, int spacedim>
293void
294setup_periodic_boundary_conditions(
295 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
296 const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
297
298{
299 // Setup periodic boundary conditions
300 for (auto const &[id, type] : boundary_conditions.type)
301 {
302 if (type == BoundaryConditions::BoundaryType::periodic)
303 {
304 auto triangulation_ptr = &triangulation;
305
306 std::vector<GridTools::PeriodicFacePair<
307 typename Triangulation<dim, spacedim>::cell_iterator>>
308 periodicity_vector;
309 GridTools::collect_periodic_faces(
310 *dynamic_cast<Triangulation<dim, spacedim> *>(triangulation_ptr),
311 id,
312 boundary_conditions.periodic_neighbor_id.at(id),
313 boundary_conditions.periodic_direction.at(id),
314 periodicity_vector);
315 triangulation.add_periodicity(periodicity_vector);
316 }
317 }
318}

320template <int dim, int spacedim>
321void
322read_mesh_and_manifolds(
323 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
324 const Parameters::Mesh &mesh_parameters,
325 const Parameters::Manifolds &manifolds_parameters,
326 const bool &restart,
327 const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
328{
329 attach_grid_to_triangulation(triangulation, mesh_parameters);
330
331 setup_periodic_boundary_conditions(triangulation, boundary_conditions);
332
333 // If the mesh is a GMSH mesh, we need to manually set the manifold id
334 // of the face to be that of the boundary
335 // ID, otherwise, we will be unable to use external manifold.
336 // This is done manually by looping through all faces and giving them a
337 // manifold id if there is a manifold associated with this number.
338 if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
339 {
340 // Gather all the manifold ids within a set
341 std::set<int> manifold_ids;
342 for (unsigned int i = 0; i < manifolds_parameters.size; ++i)
343 manifold_ids.insert(manifolds_parameters.id[i]);
344
345 // Reset all the manifolds manually and force them to zero
346 triangulation.reset_all_manifolds();
347
348 // If the parameter file forces the occurrence of manifold,
349 // loop over the faces of the triangulation. If the face of the
350 // triangulation has a boundary id which corresponds to a manifold id
351 // identified within the parameter file, then fix the manifold id of this
352 // face manually to be that of the boundary id. In the past, this was done
353 // by default for every face, but since 2023-12 this throws (rightfully)
354 // an error in deal.II
355 if (manifolds_parameters.size > 0)
356 {
357 for (const auto &face : triangulation.active_face_iterators())
358 {
359 if (face->at_boundary() &&
360 manifold_ids.find(face->boundary_id()) != manifold_ids.end())
361 face->set_all_manifold_ids(face->boundary_id());
362 }
363 }
364 }
365
366 // Finally attach the manifolds to the triangulation
367 // Right now this should only occur for GMSH mesh, but the function is generic
368 // enough.
369 attach_manifolds_to_triangulation(triangulation, manifolds_parameters);
370
371 if (mesh_parameters.simplex)
372 {
373 // Refinement isn't possible yet
374 }
375 else
376 {
377 if (mesh_parameters.refine_until_target_size)
378 {
379 double minimal_cell_size =
380 GridTools::minimal_cell_diameter(triangulation);
381 double target_size = mesh_parameters.target_size;
382 unsigned int number_refinement =
383 floor(std::log(minimal_cell_size / target_size) / std::log(2));
384 triangulation.refine_global(number_refinement);
385 }
386 else if (!restart)
387 {
388 const int initial_refinement = mesh_parameters.initial_refinement;
389 triangulation.refine_global(initial_refinement);
390 refine_triangulation_at_boundaries(
391 mesh_parameters.boundaries_to_refine,
392 mesh_parameters.initial_refinement_at_boundaries,
393 triangulation);
394 }
395 }
396}
