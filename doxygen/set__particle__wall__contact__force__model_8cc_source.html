8template <int dim, typename PropertiesIndex>
9std::shared_ptr<ParticleWallContactForce<dim, PropertiesIndex>>
10set_particle_wall_contact_force_model(
11 const DEMSolverParameters<dim> &dem_parameters,
12 const parallel::distributed::Triangulation<dim> &triangulation)
13{
14 std::shared_ptr<ParticleWallContactForce<dim, PropertiesIndex>>
15 particle_wall_contact_force_object;
16
17
18 std::vector<types::boundary_id> boundary_index =
19 triangulation.get_boundary_ids();
20 if (dem_parameters.model_parameters.particle_wall_contact_force_method ==
21 Parameters::Lagrangian::ModelParameters::ParticleWallContactForceModel::
22 linear)
23 {
24 particle_wall_contact_force_object =
25 std::make_shared<ParticleWallLinearForce<dim, PropertiesIndex>>(
26 dem_parameters, boundary_index);
27 }
28 else if (dem_parameters.model_parameters.particle_wall_contact_force_method ==
29 Parameters::Lagrangian::ModelParameters::
30 ParticleWallContactForceModel::nonlinear)
31 {
32 particle_wall_contact_force_object =
33 std::make_shared<ParticleWallNonLinearForce<dim, PropertiesIndex>>(
34 dem_parameters, boundary_index);
35 }
36 else if (dem_parameters.model_parameters.particle_wall_contact_force_method ==
37 Parameters::Lagrangian::ModelParameters::
38 ParticleWallContactForceModel::JKR)
39 {
40 particle_wall_contact_force_object =
41 std::make_shared<ParticleWallJKRForce<dim, PropertiesIndex>>(
42 dem_parameters, boundary_index);
43 }
44 else if (dem_parameters.model_parameters.particle_wall_contact_force_method ==
45 Parameters::Lagrangian::ModelParameters::
46 ParticleWallContactForceModel::DMT)
47 {
48 particle_wall_contact_force_object =
49 std::make_shared<ParticleWallDMTForce<dim, PropertiesIndex>>(
50 dem_parameters, boundary_index);
51 }
52 else
53 {
54 throw "The chosen particle-wall contact force model is invalid";
55 }
56 return particle_wall_contact_force_object;
57}
