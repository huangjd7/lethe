20/// Function to construct the Butcher's table for a given SDIRK method
21SDIRKTable
22sdirk_table(const Parameters::SimulationControl::TimeSteppingMethod method)
23{
24 SDIRKTable table;
25
26 // Important note : the nomenclature used for the name of the SDIRK methods
27 // are sdirkOrderStage sdirk22 means SDIRK with order 2 and 2 stages, sdirk33
28 // means SDIRK with order 3 and 3 stages
29 if (method == Parameters::SimulationControl::TimeSteppingMethod::sdirk22)
30 {
31 // SDIRK22 - Alexander's method (L-stable)
32 // Reference: R. Alexander, Diagonally implicit Runge–Kutta methods for
33 // stiff ODEs, SIAM Journal on Numerical Analysis, 14(6), 1006–1021, 1977.
34 // DOI: https://doi.org/10.1137/0714068
35 const double alpha = (2. - std::sqrt(2.)) / 2.;
36
37 table.A.reinit(2, 2);
38 table.A(0, 0) = alpha;
39 table.A(1, 0) = 1.0 - alpha;
40 table.A(1, 1) = alpha;
41
42 table.c.resize(2);
43 table.c[0] = alpha;
44 table.c[1] = 1.0;
45
46 table.b.resize(2);
47 table.b[0] = 1.0 - alpha;
48 table.b[1] = alpha;
49
50 return table;
51 }
52
53 // Important note : the nomenclature used for the name of the SDIRK methods
54 // are sdirkOrderStage sdirk22 means SDIRK with order 2 and 2 stages, sdirk33
55 // means SDIRK with order 3 and 3 stages
56 else if (method == Parameters::SimulationControl::TimeSteppingMethod::sdirk33)
57 {
58 // SDIRK33 - 3-stage, 3rd-order L-stable method (Kennedy-Carpenter)
59 // Reference: J. F. G. Kennedy and A. D. Carpenter, "Diagonally Implicit
60 // Runge-KuttaMethods for Ordinary Differential Equation"
61 // NASA/TM–2016–219173 pp. 77, March 2016. No DOI available URL :
62 // https://ntrs.nasa.gov/api/citations/20160005923/downloads/20160005923.pdf
63 const double gamma = 0.4358665215;
64
65 table.A.reinit(3, 3);
66 table.A(0, 0) = gamma;
67 table.A(1, 0) = 0.2820667393;
68 table.A(1, 1) = gamma;
69 table.A(2, 0) = 1 - 1.772630128 - gamma;
70 table.A(2, 1) = 1.772630128;
71 table.A(2, 2) = gamma;
72
73 table.c = {gamma, 0.7179332608, 1.0};
74 table.b = {1 - 1.772630128 - gamma, 1.772630128, gamma};
75
76 return table;
77 }
78
79 else
80 {
81 AssertThrow(false, ExcMessage("Unknown SDIRK method in sdirk_table()."));
82 return table;
83 }
84}

87// Function to extract the coefficients of a given stage from a SDIRK Butcher's

90SDIRKStageData::SDIRKStageData(const SDIRKTable &table,
91 const unsigned int stage_i)
92{
93 AssertThrow(table.c.size() == table.b.size(),
94 ExcMessage("b and c vectors must have the same size."));
95
96 AssertThrow(table.A.m() == table.b.size(),
97 ExcMessage(
98 "Matrix A dimensions must match size of b and c vectors."));
99
100 // The stages take the values 1, ..., nstages
101
102 a_ij.resize(stage_i);
103
104 for (unsigned int j = 0; j < stage_i; ++j)
105 // The Butcher's table starts at index 0 whereas the stages take the values
106 // 1, ..., nstages
107 a_ij[j] = table.A(stage_i - 1, j);
108
109 c_i = table.c[stage_i - 1];
110 b_i = table.b[stage_i - 1];
111}
