64/**
65 * @brief Refine a mesh around specific boundary ids
66 *
67 * @param[in] n_refinement The number of times the boundary should be refined
68 *
69 * @param[in] boundary_ids The boundary ids for which the cells should be
70 * refined
71 *
72 * @param[in, out] triangulation The triangulation on which refinement must be
73 * carried out
74 *
75 */
76template <int dim, int spacedim = dim>
77void
78refine_triangulation_at_boundaries(
79 const std::vector<int> boundary_ids,
80 const unsigned int n_refinement,
81 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation)
82{
83 // For the amount of refinements required
84 for (unsigned int r = 0; r < n_refinement; ++r)
85 {
86 // Loop over the cells and flag all cells which are within the list of
87 // boundary ids
88 for (const auto &cell : triangulation.active_cell_iterators())
89 {
90 if (cell->is_locally_owned())
91 {
92 for (const auto f : cell->face_indices())
93
94 {
95 if (cell->face(f)->at_boundary())
96 {
97 if (std::find(boundary_ids.begin(),
98 boundary_ids.end(),
99 cell->face(f)->boundary_id()) !=
100 boundary_ids.end())
101 {
102 cell->set_refine_flag();
103 }
104 }
105 }
106 }
107 }
108 triangulation.execute_coarsening_and_refinement();
109 }
110}
