13/**
14 * @brief Calculate the stabilization parameter for the Navier-Stokes equations
15 * in steady-state
16 * @return Value of the stabilization parameter - tau
17 *
18 * @param u_mag Magnitude of the velocity
19 *
20 * @param kinematic_viscosity Kinematic viscosity
21 *
22 * @param h Cell size; it should be calculated using the diameter of a sphere of
23 * equal volume to that of the cell.
24 */
25inline double
26calculate_navier_stokes_gls_tau_steady(const double u_mag,
27 const double kinematic_viscosity,
28 const double h)
29{
30 return 1. / std::sqrt(Utilities::fixed_power<2>(2. * u_mag / h) +
31 9 * Utilities::fixed_power<2>(4 * kinematic_viscosity /
32 (h * h)));
33}

35/**
36 * @brief Calculate the stabilization parameter for the transient Navier-Stokes
37 * equations
38 * @return Value of the stabilization parameter - tau
39 *
40 * @param u_mag Magnitude of the velocity
41 *
42 * @param kinematic_viscosity Kinematic viscosity
43 *
44 * @param h Cell size; it should be calculated using the diameter of a sphere of
45 * equal volume to that of the cell.
46 *
47 * @param sdt Inverse of the time-step (1/dt)
48 */
49
50inline double
51calculate_navier_stokes_gls_tau_transient(const double u_mag,
52 const double kinematic_viscosity,
53 const double h,
54 const double sdt)
55{
56 return 1. / std::sqrt(Utilities::fixed_power<2>(sdt) +
57 Utilities::fixed_power<2>(2. * u_mag / h) +
58 9 * Utilities::fixed_power<2>(4 * kinematic_viscosity /
59 (h * h)));
60}
