297/**
298 * @brief A solver for the incompressible Navier-Stokes equations implemented
299 * in a matrix-free fashion.
300 *
301 * A matrix-free stabilized solver for the incompressible Navier-Stokes
302 * equations implemented in a matrix-free fashion with a sum-factorization
303 * approach. It uses a continuous Galerkin discretization and solves the
304 * fully-coupled discretized problem in a monolithic way using multigrid
305 * preconditioners.
306 *
307 * @tparam dim An integer that denotes the dimension of the space in which
308 * the flow is solved.
309 */
310template <int dim>
311class FluidDynamicsMatrixFree
312 : public NavierStokesBase<dim,
313 LinearAlgebra::distributed::Vector<double>,
314 IndexSet>
315{
316 using VectorType = LinearAlgebra::distributed::Vector<double>;
317
318public:
319 /**
320 * @brief Constructor that sets the finite element degree and system operator
321 * according to simulation parameters.
322 *
323 * @param[in] nsparam Relevant parameters for the solver.
324 */
325 FluidDynamicsMatrixFree(SimulationParameters<dim> &nsparam);
326
327 /**
328 * @brief Destructor.
329 *
330 */
331 ~FluidDynamicsMatrixFree();
332
333 /**
334 * @brief Solve the problem defined by simulation parameters by iterating
335 * through time or through the mesh refinements.
336 */
337 virtual void
338 solve();
339
340protected:
341 /**
342 * @brief Setup the degrees of freedom, system constraints, system operator
343 * and solution vectors.
344 */
345 virtual void
346 setup_dofs_fd() override;
347
348 /**
349 * @brief Set the initial conditions for the solver. If the simulation is
350 * restarted from a checkpoint, the initial solution setting is bypassed
351 * and the checkpoint is instead read.
352 *
353 * @param[in] initial_condition_type The type of initial condition to be set.
354 *
355 * @param[in] restart A boolean that indicates if the simulation is being
356 * restarted. If set to true, the initial conditions are never set, but are
357 * instead overriden by the read_checkpoint functionality.
358 *
359 **/
360 virtual void
361 set_initial_condition_fd(
362 Parameters::InitialConditionType initial_condition_type,
363 bool restart = false) override;
364
365 /**
366 * @brief Assemble the matrix associated with the solver. Only required for
367 * compilation and it is not used for the matrix free solver.
368 */
369 virtual void
370 assemble_system_matrix() override;
371
372 /**
373 * @brief Assemble the system right hand side associated with the solver.
374 */
375 virtual void
376 assemble_system_rhs() override;
377
378 /**
379 * @brief Update the average velocity field solution in the multiphyscics interface.
380 */
381 virtual void
382 update_multiphysics_time_average_solution() override;
383
384
385 /**
386 * @brief Provide present and previous flow solutions to the multiphysics

390 virtual void
391 update_solutions_for_multiphysics() override;
392
393 /**
394 * @brief Calculate and store time derivatives of previous solutions according to

397 void
398 calculate_time_derivative_previous_solutions();
399
400 /**
401 * @brief Set up appropriate preconditioner.
402 *
403 */
404 void
405 setup_preconditioner() override;
406
407 /**
408 * @brief Solve the linear system of equations using the method specified in
409 * the simulation parameters.
410 *
411 * @param[in] initial_step Indicates if this is the first solution of the
412 * linear system. If this is the case, the non_zero version of the constraints
413 * are used for the Dirichlet boundary conditions.
414 *
415 * @param[in] renewed_matrix Indicates if the matrix has been reassembled, and
416 * thus the preconditioner needs to be reassembled.
417 */
418 void
419 solve_linear_system(const bool initial_step,
420 const bool renewed_matrix = true) override;
421
422private:

445 const double relative_residual);

452 setup_GMG();

494
495 /**
496 * @brief Describes the layout of DoFs for FE_Q_iso_Q1 elements.
497 *
498 */
499 DoFHandler<dim> dof_handler_fe_q_iso_q1;
500
501 /**
502 * @brief Trilinos vector storing the average velocities that are provided to other

520};
