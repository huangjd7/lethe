8template <int dim, typename PropertiesIndex>
9void
10log_collision_data(
11 const DEMSolverParameters<dim> &parameters,
12 typename DEM::dem_data_structures<dim>::particle_wall_in_contact
13 &particle_wall_pairs_in_contact,
14 const double current_time,
15 OngoingCollisionLog<dim> &ongoing_collision_log,
16 CompletedCollisionLog<dim> &collision_event_log,
17 const ConditionalOStream &pcout)
18{
19 // Looping over all the active particles in particle-wall pairs
20 for (auto &&pairs_in_contact_content :
21 particle_wall_pairs_in_contact | boost::adaptors::map_values)
22 {
23 // Iterating over a map which contains the required information for
24 // calculation of the contact force for each particle
25 for (auto &&contact_info :
26 pairs_in_contact_content | boost::adaptors::map_values)
27 {
28 // Defining local variables which will be used within the contact
29 // calculation
30 auto particle = contact_info.particle;
31 auto particle_properties = particle->get_properties();
32 Point<3> point_on_boundary = contact_info.point_on_boundary;
33 // Normal vector from the wall to the particle
34 Tensor<1, 3> normal_vector = contact_info.normal_vector;
35 // Getting particle 3d location
36 Point<3> particle_location_3d = get_location(particle);
37 // Defining a tensor which connects the point_on_boundary to the
38 // center of particle
39 Tensor<1, 3> point_to_particle_vector =
40 particle_location_3d - point_on_boundary;
41 Tensor<1, 3> projected_vector =
42 find_projection(point_to_particle_vector, normal_vector);
43 // Calculating the particle-wall distance using the projected vector
44 double normal_overlap =
45 ((particle_properties[PropertiesIndex::dp]) * 0.5) -
46 (projected_vector.norm());
47 types::boundary_id boundary_id = contact_info.boundary_id;
48
49 // If we log all walls or if the boundary ID is in the list of
50 // particle-wall collision boundary IDs, we log the collision
51 // information.
52 if (parameters.post_processing.log_collisions_with_all_walls ||
53 std::find(parameters.post_processing
54 .particle_wall_collision_boundary_ids.begin(),
55 parameters.post_processing
56 .particle_wall_collision_boundary_ids.end(),
57 boundary_id) !=
58 parameters.post_processing.particle_wall_collision_boundary_ids
59 .end())
60 {
61 unsigned int particle_id = particle->get_id();
62 if (normal_overlap > 0 &&
63 !ongoing_collision_log.is_in_collision(particle_id,
64 boundary_id))
65 {
66 collision_log<dim> start_log;
67 start_log.particle_id = particle_id;
68 start_log.velocity[0] =
69 particle_properties[PropertiesIndex::v_x];
70 start_log.velocity[1] =
71 particle_properties[PropertiesIndex::v_y];
72 start_log.velocity[2] =
73 particle_properties[PropertiesIndex::v_z];
74 start_log.omega[0] =
75 particle_properties[PropertiesIndex::omega_x];
76 start_log.omega[1] =
77 particle_properties[PropertiesIndex::omega_y];
78 start_log.omega[2] =
79 particle_properties[PropertiesIndex::omega_z];
80 start_log.time = current_time;
81 start_log.boundary_id = contact_info.boundary_id;
82 ongoing_collision_log.start_collision(
83 start_log); // Start logging the collision
84
85 // Print the start of the collision in the terminal if
86 // verbosity is set to verbose
87 if (parameters.post_processing.collision_verbosity ==
88 Parameters::Verbosity::verbose)
89 {
90 pcout << "Collision with boundary "
91 << start_log.boundary_id << " started for particle "
92 << particle_id << std::endl;
93 }
94 }
95
96 // If the particle does not have a positive overlap anymore, the
97 // collision has ended. Consequently, it now needs to be
98 // removed from ongoing_collision_log since the contact has
99 // reached its end.
100 if (normal_overlap < 0 &&
101 ongoing_collision_log.is_in_collision(particle_id,
102 boundary_id))
103 {
104 collision_log<dim> end_log;
105 end_log.particle_id = particle_id;
106 end_log.velocity[0] =
107 particle_properties[PropertiesIndex::v_x];
108 end_log.velocity[1] =
109 particle_properties[PropertiesIndex::v_y];
110 end_log.velocity[2] =
111 particle_properties[PropertiesIndex::v_z];
112 end_log.omega[0] =
113 particle_properties[PropertiesIndex::omega_x];
114 end_log.omega[1] =
115 particle_properties[PropertiesIndex::omega_y];
116 end_log.omega[2] =
117 particle_properties[PropertiesIndex::omega_z];
118 end_log.time = current_time;
119 collision_log<dim> start_log;
120
121 // End the collision for the particle and retrieve the start
122 // log
123 ongoing_collision_log.end_collision(particle_id,
124 boundary_id,
125 start_log);
126 end_log.boundary_id = start_log.boundary_id;
127 collision_event<dim> event;
128 event.particle_id = particle_id;
129 event.start_log = start_log;
130 event.end_log = end_log;
131
132 // Add the completed collision event to the collision event
133 // log
134 collision_event_log.add_event(event);
135
136 // Print the end of the collision in the terminal if verbosity
137 // is set to verbose
138 if (parameters.post_processing.collision_verbosity ==
139 Parameters::Verbosity::verbose)
140 {
141 pcout << "Collision with boundary " << end_log.boundary_id
142 << " ended for particle " << particle_id
143 << std::endl;
144 }
145 }
146 }
147 }
148 }
149}
150template <int dim>
151void
152write_collision_stats(const DEMSolverParameters<dim> &parameters,
153 const CompletedCollisionLog<dim> &collision_event_log)
154{
155 // Open a file
156 std::ofstream myfile;
157 std::string sep;
158 std::string filename = parameters.post_processing.collision_stats_file_name;
159 // Check if a .csv or .dat extension is specified in the filename, if not add
160 // ".csv"
161 std::size_t csv_file = filename.find(".csv");
162 std::size_t dat_file = filename.find(".dat");
163 if ((csv_file == std::string::npos) && (dat_file == std::string::npos))
164 filename += ".csv";
165 myfile.open(filename);
166 if (filename.substr(filename.find_last_of('.') + 1) == ".dat")
167 {
168 myfile
169 << "particle_id boundary_id start_time end_time start_particle_velocity_x start_particle_velocity_y start_particle_velocity_z start_particle_angular_velocity_x start_particle_angular_velocity_y start_particle_angular_velocity_z end_particle_velocity_x end_particle_velocity_y end_particle_velocity_z end_particle_angular_velocity_x end_particle_angular_velocity_y end_particle_angular_velocity_z"
170 << std::endl;
171 sep = " ";
172 }
173 else // .csv is default
174 {
175 myfile
176 << "particle_id,boundary_id,start_time,end_time,start_particle_velocity_x,start_particle_velocity_y,start_particle_velocity_z,start_particle_angular_velocity_x,start_particle_angular_velocity_y,start_particle_angular_velocity_z,end_particle_velocity_x,end_particle_velocity_y,end_particle_velocity_z,end_particle_angular_velocity_x,end_particle_angular_velocity_y,end_particle_angular_velocity_z"
177 << std::endl;
178 sep = ",";
179 }
180 // Write the collision statistics
181 for (const auto &event : collision_event_log.get_events())
182 {
183 const auto &start = event.start_log;
184 const auto &end = event.end_log;
185
186 // Write the collision data to the file
187 myfile << start.particle_id << sep << static_cast<int>(start.boundary_id)
188 << sep << start.time << sep << end.time << sep << start.velocity[0]
189 << sep << start.velocity[1] << sep << start.velocity[2] << sep
190 << start.omega[0] << sep << start.omega[1] << sep << start.omega[2]
191 << sep << end.velocity[0] << sep << end.velocity[1] << sep
192 << end.velocity[2] << sep << end.omega[0] << sep << end.omega[1]
193 << sep << end.omega[2] << std::endl;
194 }
195 myfile.close();
196}
