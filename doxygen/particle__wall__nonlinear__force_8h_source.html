104 /**
105 * @brief Calculation of the rolling resistance torque model using the
106 * physical properties and the constant model.
107 *
108 * @param[in] particle_properties Particle one properties.
109 * @param[in] effective_rolling_friction_coefficient Effective rolling
110 * friction coefficient.
111 * @param[in] normal_force_norm Normal force norm.
112 *
113 * @return Rolling resistance torque.
114 */
115 inline Tensor<1, 3>
116 constant_resistance(
117 const ArrayView<const double> &particle_properties,
118 const double effective_rolling_friction_coefficient,
119 const double /*effective_rolling_viscous_damping_coefficient */,
120 const double normal_force_norm,
121 const double /*dt*/,
122 const double /*normal_spring_constant*/,
123 const Tensor<1, 3> & /*normal_unit_vector*/,
124 Tensor<1, 3> & /*cumulative_rolling_resistance_spring_torque*/)
125 {
126 // Getting the angular velocity of particle in the vector format
127 Tensor<1, 3> angular_velocity;
128 for (int d = 0; d < 3; ++d)
129 {
130 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
131 }
132
133 // Calculation of particle-wall angular velocity (norm of the
134 // particle angular velocity)
135 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
136
137 double omega_value = angular_velocity.norm();
138 if (omega_value != 0)
139 {
140 particle_wall_angular_velocity = angular_velocity / omega_value;
141 }
142
143 // Calculation of rolling resistance torque
144 Tensor<1, 3> rolling_resistance_torque =
145 -effective_rolling_friction_coefficient *
146 (particle_properties[PropertiesIndex::dp] * 0.5) * normal_force_norm *
147 particle_wall_angular_velocity;
148
149 return rolling_resistance_torque;
150 }

152 /**
153 * @brief Calculation of the rolling resistance torque model using the
154 * physical properties and the viscous model.
155 *
156 * @param[in] particle_properties Particle one properties.
157 * @param[in] effective_rolling_friction_coefficient Effective rolling
158 * friction coefficient.
159 * @param[in] normal_force_norm Normal force norm.
160 * @param[in] normal_unit_vector Normal unit contact vector between the
161 * particle and the wall.
162 *
163 * @return Viscous rolling resistance torque.
164 */
165 inline Tensor<1, 3>
166 viscous_resistance(
167 const ArrayView<const double> &particle_properties,
168 const double effective_rolling_friction_coefficient,
169 const double /*effective_rolling_viscous_damping_coefficient*/,
170 const double normal_force_norm,
171 const double /*dt*/,
172 const double /*normal_spring_constant*/,
173 const Tensor<1, 3> &normal_unit_vector,
174 Tensor<1, 3> & /*cumulative_rolling_resistance_spring_torque*/)
175 {
176 // Getting the angular velocity of particle in the vector format
177 Tensor<1, 3> angular_velocity;
178 for (int d = 0; d < 3; ++d)
179 {
180 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
181 }
182
183 // Calculation of particle-wall angular velocity (norm of the
184 // particle angular velocity)
185 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
186
187 double omega_value = angular_velocity.norm();
188 if (omega_value != 0)
189 {
190 particle_wall_angular_velocity = angular_velocity / omega_value;
191 }
192
193 Tensor<1, 3> v_omega =
194 cross_product_3d(angular_velocity,
195 particle_properties[PropertiesIndex::dp] * 0.5 *
196 normal_unit_vector);
197
198 // Calculation of rolling resistance torque
199 Tensor<1, 3> rolling_resistance_torque =
200 -effective_rolling_friction_coefficient *
201 particle_properties[PropertiesIndex::dp] * 0.5 * normal_force_norm *
202 v_omega.norm() * particle_wall_angular_velocity;
203
204 return rolling_resistance_torque;
205 }

207 /**
208 * @brief Calculation of the rolling resistance torque model using the
209 * physical properties and the elastic-plastic spring-dashpot model
210 *
211 * @param[in] particle_properties Particle one properties.
212 * @param[in] effective_rolling_friction_coefficient Effective rolling
213 * friction coefficient.
214 * @param[in] effective_rolling_viscous_damping_coefficient Effective rolling
215 * viscous damping coefficient.
216 * @param[in] normal_force_norm Normal force norm.
217 * @param[in] dt DEM time step.
218 * @param[in] normal_spring_constant Normal spring constant.
219 * @param[in] normal_unit_vector Normal unit contact vector between the
220 * particle and the wall.
221 * @param[in,out] cumulative_rolling_resistance_spring_torque Cumulative
222 * rolling resistance torque for the EPSD rolling resistance model.
223 *
224 * @return Elasto-platique spring-dashpot rolling resistance torque
225 */
226 inline Tensor<1, 3>
227 epsd_resistance(const ArrayView<const double> &particle_properties,
228 const double effective_rolling_friction_coefficient,
229 const double effective_rolling_viscous_damping_coefficient,
230 const double normal_force_norm,
231 const double dt,
232 const double normal_spring_constant,
233 const Tensor<1, 3> &normal_unit_vector,
234 Tensor<1, 3> &cumulative_rolling_resistance_spring_torque)
235 {
236 // Useful value used more than once:
237 // mu_r * R_e
238 const double mu_r_times_R_e = effective_rolling_friction_coefficient * 0.5 *
239 particle_properties[PropertiesIndex::dp];
240
241 // Getting the angular velocity of particle in the vector format.
242 Tensor<1, 3>
243 omega_ij; // We ignore the rotation of the wall. j is the particle.
244 for (int d = 0; d < 3; ++d)
245 omega_ij[d] = particle_properties[PropertiesIndex::omega_x + d];
246
247 // Non-collinear component of the relative velocity.
248 const Tensor<1, 3> omega_ij_perpendicular =
249 omega_ij -
250 scalar_product(omega_ij, normal_unit_vector) * normal_unit_vector;
251
252 // Delta theta
253 const Tensor<1, 3> delta_theta = dt * omega_ij_perpendicular;
254
255 // Rolling stiffness
256 const double K_r = [&]() {
257 if constexpr (dim == 3)
258 return 2.25 * normal_spring_constant *
259 Utilities::fixed_power<2>(mu_r_times_R_e);
260 else
261 return 3. * normal_spring_constant *
262 Utilities::fixed_power<2>(mu_r_times_R_e);
263 }();
264
265 // Update the spring torque
266 cumulative_rolling_resistance_spring_torque -= K_r * delta_theta;
267
268 // Limiting spring torque
269 const double M_r_max = mu_r_times_R_e * normal_force_norm;
270
271 const double rolling_resistance_spring_torque_norm =
272 cumulative_rolling_resistance_spring_torque.norm();
273
274 // Total inertia of particle i evaluated at its surface. Mass of the
275 // wall is considered infinite, thus I_i is equal to I_e. (Effective
276 // inertia)
277 const double I_e =
278 1.4 * particle_properties[PropertiesIndex::mass] *
279 Utilities::fixed_power<2>(0.5 * particle_properties[PropertiesIndex::dp]);
280
281 // C_r_crit = 2. * sqrt(I_r * K_r)
282 // C_r = eta_r * C_r_crit
283 const double C_r =
284 effective_rolling_viscous_damping_coefficient * 2. * sqrt(I_e * K_r);
285
286 // Similarly to the coulomb limit, the spring torque must be decrease to the
287 // limit value if it exceeds the limiting spring toque.
288 if (rolling_resistance_spring_torque_norm > M_r_max)
289 {
290 cumulative_rolling_resistance_spring_torque =
291 cumulative_rolling_resistance_spring_torque *
292 (M_r_max / rolling_resistance_spring_torque_norm);
293
294 // If the limiting spring torque is exceeded, the damping is multiplied
295 // by the f_coefficient which should be between 0 and 1.
296
297 return cumulative_rolling_resistance_spring_torque -
298 f_coefficient_epsd * C_r * omega_ij_perpendicular;
299 }
300 // Minus sign has been verified
301 return cumulative_rolling_resistance_spring_torque -
302 C_r * omega_ij_perpendicular;
303 }
