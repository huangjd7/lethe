86 /**
87 * @brief Carries out calculation of the rolling resistance torque using the no
88 * resistance model
89 *

92 * @param normal_force_norm Normal force norm

95 inline Tensor<1, 3>
96 no_resistance(const ArrayView<const double> & /*particle_properties*/,
97 const double /*effective_rolling_friction_coefficient*/,
98 const double /*normal_force_norm*/,

105 /**
106 * @brief Carries out calculation of the rolling resistance torque using the
107 * constant model
108 *
109 * @param particle_properties Particle properties
110 * @param effective_rolling_friction_coefficient Effective rolling friction
111 * coefficient
112 * @param normal_force_norm Normal force norm
113 * @return rolling resistance torque
114 */
115 inline Tensor<1, 3>
116 constant_resistance(const ArrayView<const double> &particle_properties,
117 const double effective_rolling_friction_coefficient,
118 const double normal_force_norm,
119 const Tensor<1, 3> & /*normal_contact_vector*/)
120 {
121 // Getting the angular velocity of particle in the vector format
122 Tensor<1, 3> angular_velocity;
123 for (int d = 0; d < 3; ++d)
124 {
125 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
126 }
127
128 // Calculation of particle-wall angular velocity (norm of the
129 // particle angular velocity)
130 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
131
132 double omega_value = angular_velocity.norm();
133 if (omega_value != 0)
134 {
135 particle_wall_angular_velocity = angular_velocity / omega_value;
136 }
137
138 // Calculation of rolling resistance torque
139 Tensor<1, 3> rolling_resistance_torque =
140 -effective_rolling_friction_coefficient *
141 (particle_properties[PropertiesIndex::dp] * 0.5) * normal_force_norm *
142 particle_wall_angular_velocity;
143
144 return rolling_resistance_torque;
145 }

147 /**
148 * @brief Carries out calculation of the rolling resistance torque using the
149 * viscous model
150 *
151 * @param particle_properties Particle properties
152 * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
153 * @param normal_force_norm Normal force norm
154 * @return rolling resistance torque
155 */
156 inline Tensor<1, 3>
157 viscous_resistance(const ArrayView<const double> &particle_properties,
158 const double effective_rolling_friction_coefficient,
159 const double normal_force_norm,
160 const Tensor<1, 3> &normal_contact_vector)
161 {
162 // Getting the angular velocity of particle in the vector format
163 Tensor<1, 3> angular_velocity;
164 for (int d = 0; d < 3; ++d)
165 {
166 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
167 }
168
169 // Calculation of particle-wall angular velocity (norm of the
170 // particle angular velocity)
171 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
172
173 double omega_value = angular_velocity.norm();
174 if (omega_value != 0)
175 {
176 particle_wall_angular_velocity = angular_velocity / omega_value;
177 }
178
179 Tensor<1, 3> v_omega =
180 cross_product_3d(angular_velocity,
181 particle_properties[PropertiesIndex::dp] * 0.5 *
182 normal_contact_vector);
183
184 // Calculation of rolling resistance torque
185 Tensor<1, 3> rolling_resistance_torque =
186 -effective_rolling_friction_coefficient *
187 particle_properties[PropertiesIndex::dp] * 0.5 * normal_force_norm *
188 v_omega.norm() * particle_wall_angular_velocity;
189
190 return rolling_resistance_torque;
191 }

193 /**
194 * @brief Initialize every particle-wall properties required
195 * for the force calculation. This function is required for the DMT force
196 * model, since the dmt_cut_off_distance attribute needs the be set to const.
197 * To do so, the effective surface energy and of the effective hamaker contant
198 * need to be initialized before hand. For code clarity, we also initialized
199 * every other parameter, otherwise the initialization would take place at two
200 * different places in the code.
201 *
202 * @param[in] dem_parameters DEM parameters declared in the .prm file
203 */
204 void
205 initialize_particle_wall_properties(
206 const DEMSolverParameters<dim> &dem_parameters)
207 {
208 // Wall properties
209 const double wall_youngs_modulus =
210 dem_parameters.lagrangian_physical_properties.youngs_modulus_wall;
211 const double wall_poisson_ratio =
212 dem_parameters.lagrangian_physical_properties.poisson_ratio_wall;
213 const double wall_restitution_coefficient =
214 dem_parameters.lagrangian_physical_properties
215 .restitution_coefficient_wall;
216 const double wall_friction_coefficient =
217 dem_parameters.lagrangian_physical_properties.friction_coefficient_wall;
218 const double wall_rolling_friction_coefficient =
219 dem_parameters.lagrangian_physical_properties.rolling_friction_wall;
220 const double wall_rolling_viscous_damping =
221 dem_parameters.lagrangian_physical_properties
222 .rolling_viscous_damping_wall;
223 const double wall_surface_energy =
224 dem_parameters.lagrangian_physical_properties.surface_energy_wall;
225 const double wall_hamaker_constant =
226 dem_parameters.lagrangian_physical_properties.hamaker_constant_wall;
227 for (unsigned int i = 0;
228 i < dem_parameters.lagrangian_physical_properties.particle_type_number;
229 ++i)
230 {
231 // Particle properties
232 const double particle_youngs_modulus =
233 dem_parameters.lagrangian_physical_properties.youngs_modulus_particle
234 .at(i);
235 const double particle_poisson_ratio =
236 dem_parameters.lagrangian_physical_properties.poisson_ratio_particle
237 .at(i);
238 const double particle_restitution_coefficient =
239 dem_parameters.lagrangian_physical_properties
240 .restitution_coefficient_particle.at(i);
241 const double particle_friction_coefficient =
242 dem_parameters.lagrangian_physical_properties
243 .friction_coefficient_particle.at(i);
244 const double particle_rolling_friction_coefficient =
245 dem_parameters.lagrangian_physical_properties
246 .rolling_friction_coefficient_particle.at(i);
247 const double particle_rolling_viscous_damping_coefficient =
248 dem_parameters.lagrangian_physical_properties
249 .rolling_viscous_damping_coefficient_particle.at(i);
250 const double particle_surface_energy =
251 dem_parameters.lagrangian_physical_properties.surface_energy_particle
252 .at(i);
253 const double particle_hamaker_constant =
254 dem_parameters.lagrangian_physical_properties
255 .hamaker_constant_particle.at(i);
256
257
258 // Effective particle-wall properties.
259 this->effective_youngs_modulus[i] =
260 (particle_youngs_modulus * wall_youngs_modulus) /
261 (wall_youngs_modulus *
262 (1 - particle_poisson_ratio * particle_poisson_ratio) +
263 particle_youngs_modulus *
264 (1 - wall_poisson_ratio * wall_poisson_ratio) +
265 DBL_MIN);
266
267 this->effective_shear_modulus[i] =
268 (particle_youngs_modulus * wall_youngs_modulus) /
269 ((2 * wall_youngs_modulus * (2 - particle_poisson_ratio) *
270 (1 + particle_poisson_ratio)) +
271 (2 * particle_youngs_modulus * (2 - wall_poisson_ratio) *
272 (1 + wall_poisson_ratio)) +
273 DBL_MIN);
274
275 this->effective_coefficient_of_restitution[i] =
276 2 * particle_restitution_coefficient * wall_restitution_coefficient /
277 (particle_restitution_coefficient + wall_restitution_coefficient +
278 DBL_MIN);
279
280 const double log_coeff_restitution =
281 log(this->effective_coefficient_of_restitution[i]);
282 this->model_parameter_beta[i] =
283 log_coeff_restitution /
284 sqrt((log_coeff_restitution * log_coeff_restitution) + 9.8696);
285
286 this->effective_coefficient_of_friction[i] =
287 2 * particle_friction_coefficient * wall_friction_coefficient /
288 (particle_friction_coefficient + wall_friction_coefficient + DBL_MIN);
289
290 this->effective_coefficient_of_rolling_friction[i] =
291 2 * particle_rolling_friction_coefficient *
292 wall_rolling_friction_coefficient /
293 (particle_rolling_friction_coefficient +
294 wall_rolling_friction_coefficient + DBL_MIN);
295
296 this->effective_coefficient_of_rolling_viscous_damping[i] =
297 2 * particle_rolling_viscous_damping_coefficient *
298 wall_rolling_viscous_damping /
299 (particle_rolling_viscous_damping_coefficient +
300 wall_rolling_viscous_damping + DBL_MIN);
301
302 this->effective_surface_energy[i] =
303 particle_surface_energy + wall_surface_energy -
304 std::pow(std::sqrt(particle_surface_energy) -
305 std::sqrt(wall_surface_energy),
306 2);
307 this->effective_hamaker_constant[i] =
308 0.5 * (particle_hamaker_constant + wall_hamaker_constant);
309 }
310 }

313 /**
314 * @brief This function uses the effective surface energy and effective hamaker
315 * constant to compute the cut off distance for the force calculation.
316 *
317 * @param[in] dem_parameters DEM parameters declared in the .prm file
318 *
319 * @return Cut off distance for the force calculation.
320 */
321 double
322 set_dmt_cut_off_distance()
323 {
324 // We are searching for the large delta_o, thus the smallest effective
325 // surface energy and largest Hamaker constant.
326 double max_effective_hamaker_constant =
327 *(std::max_element(this->effective_hamaker_constant.begin(),
328 this->effective_hamaker_constant.end()));
329 double min_effective_surface_energy =
330 *(std::min_element(this->effective_surface_energy.begin(),
331 this->effective_surface_energy.end()));
332
333 const double delta_0 =
334 -std::sqrt(max_effective_hamaker_constant /
335 (12. * M_PI * min_effective_surface_energy));
336
337 return delta_0 / std::sqrt(dmt_cut_off_threshold);
338 };
339
340 // Model parameter
341 const double dmt_cut_off_threshold;
342};
