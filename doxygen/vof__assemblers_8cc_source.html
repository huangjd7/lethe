10template <int dim>
11void
12VOFAssemblerCore<dim>::assemble_matrix(const VOFScratchData<dim> &scratch_data,
13 StabilizedMethodsCopyData &copy_data)
14{
15 // Scheme and physical properties
16 const auto method = this->simulation_control->get_assembly_method();
17
18 // Loop and quadrature information
19 const auto &JxW_vec = scratch_data.JxW;
20 const unsigned int n_q_points = scratch_data.n_q_points;
21 const unsigned int n_dofs = scratch_data.n_dofs;
22 const double h = scratch_data.cell_size;
23
24 // Time steps and inverse time steps which is used for stabilization constant
25 std::vector<double> time_steps_vector =
26 this->simulation_control->get_time_steps_vector();
27 const double dt = time_steps_vector[0];
28 const double sdt = 1. / dt;
29
30 // Add a small diffusivity, used to artificially diffuse VOF
31 const double diffusivity = this->vof_parameters.diffusivity;
32
33 // Copy data elements
34 auto &strong_jacobian_vec = copy_data.strong_jacobian;
35 auto &local_matrix = copy_data.local_matrix;
36
37 // Assemble local matrix and right-hand side
38 for (unsigned int q = 0; q < n_q_points; ++q)
39 {
40 // Gather into local variables the relevant fields
41 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
42 const double velocity_divergence = scratch_data.velocity_divergences[q];
43
44 // Store JxW in local variable for faster access
45 const double JxW = JxW_vec[q];
46
47 // Define tolerance to avoid division by zero
48 const double tolerance = 1e-12;
49
50 // Compute the velocity magnitude for the stabilization parameter and the
51 // compression term for the phase indicator
52 const double u_mag = std::max(velocity.norm(), tolerance);
53
54 // Calculation of the GLS stabilization parameter. The
55 // stabilization parameter used is different if the simulation is
56 // steady or unsteady. In the unsteady case, it includes the value
57 // of the time-step. Hypothesis: advection dominated problem
58 // (Pe>3) [Bochev et al., Stability of the SUPG finite element
59 // method for transient advection-diffusion problems, CMAME 2004]
60 const double tau =
61 is_steady(method) ?
62 1. / std::sqrt(
63 Utilities::fixed_power<2>(2. * u_mag / h) +
64 9 * Utilities::fixed_power<2>(4 * diffusivity / (h * h))) :
65 1. /
66 std::sqrt(Utilities::fixed_power<2>(sdt) +
67 Utilities::fixed_power<2>(2. * u_mag / h) +
68 9 * Utilities::fixed_power<2>(4 * diffusivity / (h * h)));
69
70 for (unsigned int j = 0; j < n_dofs; ++j)
71 {
72 const auto phi_phase_j = scratch_data.phi[q][j];
73 const auto grad_phi_phase_j = scratch_data.grad_phi[q][j];
74 const auto laplacian_phi_phase_j = scratch_data.laplacian_phi[q][j];
75
76 // Strong Jacobian associated with the GLS stabilization
77 strong_jacobian_vec[q][j] +=
78 velocity * grad_phi_phase_j - diffusivity * laplacian_phi_phase_j;
79
80 if (compressible)
81 strong_jacobian_vec[q][j] += phi_phase_j * velocity_divergence;
82 }
83
84 for (unsigned int i = 0; i < n_dofs; ++i)
85 {
86 const auto phi_phase_i = scratch_data.phi[q][i];
87 const auto grad_phi_phase_i = scratch_data.grad_phi[q][i];
88
89
90 for (unsigned int j = 0; j < n_dofs; ++j)
91 {
92 const auto phi_phase_j = scratch_data.phi[q][j];
93 const auto grad_phi_phase_j = scratch_data.grad_phi[q][j];
94
95 // Weak form for advection-diffusion:
96 // u * grad(phase) + phase * grad(u) - diffusivity *
97 // laplacian(phase) = 0
98 local_matrix(i, j) +=
99 (phi_phase_i * velocity * grad_phi_phase_j +
100 diffusivity * grad_phi_phase_i * grad_phi_phase_j) *
101 JxW;
102
103 // Add compressibility term if the VOF is compressible
104 if (compressible)
105 local_matrix(i, j) +=
106 phi_phase_i * phi_phase_j * velocity_divergence * JxW;
107
108 // Addition to the cell matrix for GLS stabilization
109 local_matrix(i, j) += tau * strong_jacobian_vec[q][j] *
110 (grad_phi_phase_i * velocity) * JxW;
111 }
112 }
113 } // end loop on quadrature points
114}

117template <int dim>
118void
119VOFAssemblerCore<dim>::assemble_rhs(const VOFScratchData<dim> &scratch_data,
120 StabilizedMethodsCopyData &copy_data)
121{
122 // Scheme and physical properties
123 const auto method = this->simulation_control->get_assembly_method();
124
125 // Add a small diffusivity, used to artificially diffuse VOF
126 const double diffusivity = this->vof_parameters.diffusivity;
127
128 // Loop and quadrature information
129 const auto &JxW_vec = scratch_data.JxW;
130 const unsigned int n_q_points = scratch_data.n_q_points;
131 const unsigned int n_dofs = scratch_data.n_dofs;
132 const double h = scratch_data.cell_size;
133
134 // Time steps and inverse time steps which is used for stabilization constant
135 std::vector<double> time_steps_vector =
136 this->simulation_control->get_time_steps_vector();
137 const double dt = time_steps_vector[0];
138 const double sdt = 1. / dt;
139
140 // Copy data elements
141 auto &strong_residual_vec = copy_data.strong_residual;
142 auto &local_rhs = copy_data.local_rhs;
143
144 // Assemble local matrix and right-hand side
145 for (unsigned int q = 0; q < n_q_points; ++q)
146 {
147 // Gather into local variables the relevant fields
148 const double phase = scratch_data.present_phase_values[q];
149 const Tensor<1, dim> phase_gradient = scratch_data.phase_gradients[q];
150 const double phase_laplacians = scratch_data.phase_laplacians[q];
151 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
152 const double velocity_divergence = scratch_data.velocity_divergences[q];
153
154 // Store JxW in local variable for faster access;
155 const double JxW = JxW_vec[q];
156
157 // Define tolerance to avoid division by zero
158 const double tolerance = 1e-12;
159
160 // Compute the velocity magnitude for the stabilization parameter
161 const double u_mag = std::max(velocity.norm(), tolerance);
162
163 // Calculation of the GLS stabilization parameter. The
164 // stabilization parameter used is different if the simulation is
165 // steady or unsteady. In the unsteady case, it includes the value
166 // of the time-step. Hypothesis: advection dominated problem
167 // (Pe>3) [Bochev et al., Stability of the SUPG finite element
168 // method for transient advection-diffusion problems, CMAME 2004]
169 const double tau =
170 is_steady(method) ?
171 1. / std::sqrt(
172 Utilities::fixed_power<2>(2. * u_mag / h) +
173 9 * Utilities::fixed_power<2>(4 * diffusivity / (h * h))) :
174 1. /
175 std::sqrt(Utilities::fixed_power<2>(sdt) +
176 Utilities::fixed_power<2>(2. * u_mag / h) +
177 9 * Utilities::fixed_power<2>(4 * diffusivity / (h * h)));
178
179 // Calculate the strong residual for GLS stabilization
180 strong_residual_vec[q] +=
181 velocity * phase_gradient - diffusivity * phase_laplacians;
182
183 if (compressible)
184 strong_residual_vec[q] += phase * velocity_divergence;
185
186 for (unsigned int i = 0; i < n_dofs; ++i)
187 {
188 const auto phi_phase_i = scratch_data.phi[q][i];
189 const auto grad_phi_phase_i = scratch_data.grad_phi[q][i];
190
191 // rhs for: u * grad(phase) + phase * grad(u) - diffusivity *
192 // laplacian(phase) = 0
193 local_rhs(i) -= (phi_phase_i * velocity * phase_gradient +
194 diffusivity * grad_phi_phase_i * phase_gradient) *
195 JxW;
196
197 if (compressible)
198 local_rhs(i) -= phi_phase_i * phase * velocity_divergence * JxW;
199
200 local_rhs(i) -=
201 tau * (strong_residual_vec[q] * (grad_phi_phase_i * velocity)) *
202 JxW;
203 }
204 }
205}

211template <int dim>
212void
213VOFAssemblerBDF<dim>::assemble_matrix(const VOFScratchData<dim> &scratch_data,
214 StabilizedMethodsCopyData &copy_data)
215{
216 // Loop and quadrature information
217 const auto &JxW = scratch_data.JxW;
218 const unsigned int n_q_points = scratch_data.n_q_points;
219 const unsigned int n_dofs = scratch_data.n_dofs;
220
221 // Copy data elements
222 auto &strong_jacobian = copy_data.strong_jacobian;
223 auto &strong_residual = copy_data.strong_residual;
224 auto &local_matrix = copy_data.local_matrix;
225
226 // Time stepping information
227 const auto method = this->simulation_control->get_assembly_method();
228 // Vector for the BDF coefficients
229 const Vector<double> &bdf_coefs =
230 this->simulation_control->get_bdf_coefficients();
231 std::vector<double> phase_value(1 + number_of_previous_solutions(method));
232
233 // Loop over the quadrature points
234 for (unsigned int q = 0; q < n_q_points; ++q)
235 {
236 phase_value[0] = scratch_data.present_phase_values[q];
237
238 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
239 phase_value[p + 1] = scratch_data.previous_phase_values[p][q];
240
241
242 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
243 ++p)
244 {
245 strong_residual[q] += (bdf_coefs[p] * phase_value[p]);
246 }
247
248 for (unsigned int j = 0; j < n_dofs; ++j)
249 {
250 strong_jacobian[q][j] += bdf_coefs[0] * scratch_data.phi[q][j];
251 }
252
253
254 for (unsigned int i = 0; i < n_dofs; ++i)
255 {
256 const auto phi_phase_i = scratch_data.phi[q][i];
257 for (unsigned int j = 0; j < n_dofs; ++j)
258 {
259 const auto phi_phase_j = scratch_data.phi[q][j];
260
261 local_matrix(i, j) +=
262 phi_phase_j * phi_phase_i * bdf_coefs[0] * JxW[q];
263 }
264 }
265 }
266}

276 const unsigned int n_dofs = scratch_data.n_dofs;
277
278 // Copy data elements

280 auto &local_rhs = copy_data.local_rhs;
281
282 // Time stepping information

284 // Vector for the BDF coefficients

287 std::vector<double> phase_value(1 + number_of_previous_solutions(method));
288
289 // Loop over the quadrature points
290 for (unsigned int q = 0; q < n_q_points; ++q)
291 {
292 phase_value[0] = scratch_data.present_phase_values[q];
293
294 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
295 phase_value[p + 1] = scratch_data.previous_phase_values[p][q];
296
297 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
298 ++p)
299 {
300 strong_residual[q] += (bdf_coefs[p] * phase_value[p]);
301 }
302
303 for (unsigned int i = 0; i < n_dofs; ++i)
304 {
305 const double phi_phase_i = scratch_data.phi[q][i];
306 double local_rhs_i = 0;
307 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
308 ++p)
309 {
310 local_rhs_i -= bdf_coefs[p] * (phase_value[p] * phi_phase_i);
311 }
312 local_rhs(i) += local_rhs_i * JxW[q];
313 }
314 }

321template <int dim>
322void
323VOFAssemblerDCDDStabilization<dim>::assemble_matrix(
324 const VOFScratchData<dim> &scratch_data,
325 StabilizedMethodsCopyData &copy_data)
326{
327 // Loop and quadrature information
328 const auto &JxW_vec = scratch_data.JxW;
329 const unsigned int n_q_points = scratch_data.n_q_points;
330 const unsigned int n_dofs = scratch_data.n_dofs;
331 const double h = scratch_data.cell_size;
332
333 // Copy data elements
334 auto &local_matrix = copy_data.local_matrix;
335
336 // Assemble local matrix
337 for (unsigned int q = 0; q < n_q_points; ++q)
338 {
339 // Gather into local variables the relevant fields for faster access
340 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
341 const double JxW = JxW_vec[q];
342
343 // We use the previous phase gradient for the shock capture (explicit)
344 const Tensor<1, dim> previous_phase_gradient =
345 scratch_data.previous_phase_gradients[q];
346
347 // Define a tolerance to avoid division by zero
348 const double tolerance = 1e-12;
349
350 // Compute the velocity magnitude and phase gradient norm for the
351 // stabilization term
352 const double velocity_norm = velocity.norm();
353 const double phase_gradient_norm = previous_phase_gradient.norm();
354
355 // In Tezduyar 2003, this is denoted r in equation (67).
356 Tensor<1, dim> gradient_unit_vector =
357 previous_phase_gradient / (phase_gradient_norm + tolerance);
358
359 // We remove the diffusion aligned with the velocity as is done in the
360 // original article. In Tezduyar 2003, this is denoted s in equation (67).
361 Tensor<1, dim> velocity_unit_vector =
362 velocity / (velocity_norm + tolerance);
363
364 // Directionality tensor describing [rr - (r⋅s)^2*ss]
365 const Tensor<2, dim> dir_tensor =
366 outer_product(gradient_unit_vector, gradient_unit_vector) -
367 ((gradient_unit_vector * velocity_unit_vector) *
368 outer_product(velocity_unit_vector, velocity_unit_vector));
369
370 // Compute the artificial viscosity of the shock capture
371 const double nu_dcdd =
372 (0.5 * h * h) * velocity.norm() * phase_gradient_norm;
373
374 // Assemble DCDD shock capturing contributions
375 for (unsigned int i = 0; i < n_dofs; ++i)
376 {
377 const auto grad_phi_phase_i = scratch_data.grad_phi[q][i];
378
379 for (unsigned int j = 0; j < n_dofs; ++j)
380 {
381 const auto grad_phi_phase_j = scratch_data.grad_phi[q][j];
382 local_matrix(i, j) +=
383 (nu_dcdd * scalar_product(grad_phi_phase_i,
384 dir_tensor * grad_phi_phase_j)) *
385 JxW;
386 }
387 }
388 }
389}
