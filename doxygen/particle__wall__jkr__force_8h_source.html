105 /**
106 * @brief Calculation of the rolling resistance torque model using the
107 * physical properties and the constant model.
108 *
109 * @param[in] particle_properties Particle one properties.
110 * @param[in] effective_rolling_friction_coefficient Effective rolling
111 * friction coefficient.
112 * @param[in] normal_force_norm Normal force norm.
113 * particle and the wall.
114 *
115 * @return Rolling resistance torque
116 */
117 inline Tensor<1, 3>
118 constant_resistance(
119 const ArrayView<const double> &particle_properties,
120 const double effective_rolling_friction_coefficient,
121 const double /*effective_rolling_viscous_damping_coefficient */,
122 const double normal_force_norm,
123 const double /*dt*/,
124 const double /*normal_spring_constant*/,
125 const Tensor<1, 3> & /*normal_unit_vector*/,
126 Tensor<1, 3> & /*cumulative_rolling_resistance_spring_torque*/)
127 {
128 // Getting the angular velocity of particle in the vector format
129 Tensor<1, 3> angular_velocity;
130 for (int d = 0; d < 3; ++d)
131 {
132 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
133 }
134
135 // Calculation of particle-wall angular velocity (norm of the
136 // particle angular velocity)
137 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
138
139 double omega_value = angular_velocity.norm();
140 if (omega_value != 0)
141 {
142 particle_wall_angular_velocity = angular_velocity / omega_value;
143 }
144
145 // Calculation of rolling resistance torque
146 Tensor<1, 3> rolling_resistance_torque =
147 -effective_rolling_friction_coefficient *
148 (particle_properties[PropertiesIndex::dp] * 0.5) * normal_force_norm *
149 particle_wall_angular_velocity;
150
151 return rolling_resistance_torque;
152 }

154 /**
155 * @brief Calculation of the rolling resistance torque model using the
156 * physical properties and the viscous model.
157 *
158 * @param[in] particle_properties Particle one properties.
159 * @param[in] effective_rolling_friction_coefficient Effective rolling
160 * friction coefficient. viscous damping coefficient.
161 * @param[in] normal_force_norm Normal force norm.
162 * @param[in] normal_unit_vector Normal unit contact vector between the
163 * particle and the wall.
164 *
165 * @return rolling resistance torque
166 */
167 inline Tensor<1, 3>
168 viscous_resistance(
169 const ArrayView<const double> &particle_properties,
170 const double effective_rolling_friction_coefficient,
171 const double /*effective_rolling_viscous_damping_coefficient*/,
172 const double normal_force_norm,
173 const double /*dt*/,
174 const double /*normal_spring_constant*/,
175 const Tensor<1, 3> &normal_unit_vector,
176 Tensor<1, 3> & /*cumulative_rolling_resistance_spring_torque*/)
177 {
178 // Getting the angular velocity of particle in the vector format
179 Tensor<1, 3> angular_velocity;
180 for (int d = 0; d < 3; ++d)
181 {
182 angular_velocity[d] = particle_properties[PropertiesIndex::omega_x + d];
183 }
184
185 // Calculation of particle-wall angular velocity (norm of the
186 // particle angular velocity)
187 Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
188
189 double omega_value = angular_velocity.norm();
190 if (omega_value != 0.)
191 {
192 particle_wall_angular_velocity = angular_velocity / omega_value;
193 }
194
195 Tensor<1, 3> v_omega =
196 cross_product_3d(angular_velocity,
197 particle_properties[PropertiesIndex::dp] * 0.5 *
198 normal_unit_vector);
199
200 // Calculation of rolling resistance torque
201 Tensor<1, 3> rolling_resistance_torque =
202 -effective_rolling_friction_coefficient *
203 particle_properties[PropertiesIndex::dp] * 0.5 * normal_force_norm *
204 v_omega.norm() * particle_wall_angular_velocity;
205
206 return rolling_resistance_torque;
207 }

209 /**
210 * @brief Calculation of the rolling resistance torque model using the
211 * physical properties and the elastic-plastic spring-dashpot model
212 *
213 * @param[in] particle_properties Particle one properties.
214 * @param[in] effective_rolling_friction_coefficient Effective rolling
215 * friction coefficient.
216 * @param[in] effective_rolling_viscous_damping_coefficient Effective rolling
217 * viscous damping coefficient.
218 * @param[in] normal_force_norm Norm of the normal force tensor.
219 * @param[in] dt DEM time step.
220 * @param[in] normal_spring_constant Normal spring constant.
221 * @param[in] normal_unit_vector Normal unit contact vector between the
222 * particle and the wall.
223 * @param[in,out] cumulative_rolling_resistance_spring_torque cumulative
224 * rolling resistance torque for the EPSD rolling resistance model.
225 *
226 * @return rolling resistance torque
227 */
228 inline Tensor<1, 3>
229 epsd_resistance(const ArrayView<const double> &particle_properties,
230 const double effective_rolling_friction_coefficient,
231 const double effective_rolling_viscous_damping_coefficient,
232 const double normal_force_norm,
233 const double dt,
234 const double normal_spring_constant,
235 const Tensor<1, 3> &normal_unit_vector,
236 Tensor<1, 3> &cumulative_rolling_resistance_spring_torque)
237 {
238 // Useful value used more than once:
239 // mu_r * R_e
240 const double mu_r_times_R_e = effective_rolling_friction_coefficient * 0.5 *
241 particle_properties[PropertiesIndex::dp];
242
243 // Getting the angular velocity of particle in the vector format.
244 Tensor<1, 3>
245 omega_ij; // We ignore the rotation of the wall. j is the particle.
246 for (int d = 0; d < 3; ++d)
247 omega_ij[d] = particle_properties[PropertiesIndex::omega_x + d];
248
249
250 // Non-collinear component of the relative velocity.
251 const Tensor<1, 3> omega_ij_perpendicular =
252 omega_ij -
253 scalar_product(omega_ij, normal_unit_vector) * normal_unit_vector;
254
255 // Delta theta
256 const Tensor<1, 3> delta_theta = dt * omega_ij_perpendicular;
257
258 // Rolling stiffness
259 const double K_r = [&]() {
260 if constexpr (dim == 3)
261 return 2.25 * normal_spring_constant *
262 Utilities::fixed_power<2>(mu_r_times_R_e);
263 else
264 return 3. * normal_spring_constant *
265 Utilities::fixed_power<2>(mu_r_times_R_e);
266 }();
267
268 // Update the spring torque
269 cumulative_rolling_resistance_spring_torque -= K_r * delta_theta;
270
271 // Limiting spring torque
272 const double M_r_max = mu_r_times_R_e * normal_force_norm;
273
274 const double rolling_resistance_spring_torque_norm =
275 cumulative_rolling_resistance_spring_torque.norm();
276
277 // Total inertia of particle i is evaluated at its surface. Mass of the
278 // wall is considered infinite, thus I_i is equal to I_e. (Effective
279 // inertia)
280 const double I_e =
281 1.4 * particle_properties[PropertiesIndex::mass] *
282 Utilities::fixed_power<2>(0.5 * particle_properties[PropertiesIndex::dp]);
283
284 // C_r_crit = 2. * sqrt(I_r * K_r)
285 // C_r = eta_r * C_r_crit
286 const double C_r =
287 effective_rolling_viscous_damping_coefficient * 2. * sqrt(I_e * K_r);
288
289 // Similarly to the Coulomb limit, the spring torque must be decrease to the
290 // limit value if it exceeds the limiting spring toque.
291 if (rolling_resistance_spring_torque_norm > M_r_max)
292 {
293 cumulative_rolling_resistance_spring_torque =
294 cumulative_rolling_resistance_spring_torque *
295 (M_r_max / rolling_resistance_spring_torque_norm);
296
297 // If the limiting spring torque is exceeded, the damping is multiplied
298 // by the f_coefficient which should be between 0 and 1.
299
300 return cumulative_rolling_resistance_spring_torque -
301 f_coefficient_epsd * C_r * omega_ij_perpendicular;
302 }
303 // Minus sign has been verified
304 return cumulative_rolling_resistance_spring_torque -
305 C_r * omega_ij_perpendicular;
306 }
