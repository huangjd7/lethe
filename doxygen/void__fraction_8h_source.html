32/**
33 * @brief Calculate the area of intersection between a circular (2D) particle and a circle.
34 *
35 * @param[in] r_particle Radius of the particle
36 *
37 * @param[in] r_circle Radius of the circle
38 *
39 * @param[in] neighbor_distance Distance between the particle and the circle
40 */
41inline double
42particle_circle_intersection_2d(double r_particle,
43 double r_circle,
44 double neighbor_distance)
45{
46 return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
47 cos((pow(neighbor_distance, 2) +
48 pow(r_particle, 2) - pow(r_circle, 2)) /
49 (2 * neighbor_distance * r_particle))) +
50 Utilities::fixed_power<2, double>(r_circle) *
51 Utilities::fixed_power<-1, double>(
52 cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
53 pow(r_circle, 2)) /
54 (2 * neighbor_distance * r_circle))) -
55 0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
56 (neighbor_distance + r_particle - r_circle) *
57 (neighbor_distance - r_particle + r_circle) *
58 (neighbor_distance + r_particle + r_circle));
59}

61/**
62 * @brief Calculate the volume of intersection between a spherical (3D) particle and a sphere.
63 *
64 * @param[in] r_particle Radius of the particle
65 *
66 * @param[in] r_sphere Radius of the sphere
67 *
68 * @param[in] neighbor_distance Distance between the particle and the sphere
69 */
70
71inline double
72particle_sphere_intersection_3d(double r_particle,
73 double r_sphere,
74 double neighbor_distance)
75{
76 return M_PI *
77 Utilities::fixed_power<2, double>(r_sphere + r_particle -
78 neighbor_distance) *
79 (Utilities::fixed_power<2, double>(neighbor_distance) +
80 (2 * neighbor_distance * r_particle) -
81 (3 * Utilities::fixed_power<2, double>(r_particle)) +
82 (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
83 (3 * Utilities::fixed_power<2, double>(r_sphere))) /
84 (12 * neighbor_distance);
85}

87/**
88 * @brief Void fraction calculator. This class manages the calculation of the
89 * void fraction which is used as an auxiliary field for the solvers that solve
90 * the Volume-Averaged Navier-Stokes equations. The present architecture
91 * of the class support all of the different void fraction calculation methods
92 * within a single class instead of building a class hierarchy. This is
93 * because there are only a few void fraction calculation strategies.
94 *
95 * @tparam dim An integer that denotes the number of spatial dimensions.
96 */
97template <int dim>
98class VoidFractionBase : public PhysicsLinearSubequationsSolver
99{
100public:
101 /**
102 * @brief Constructor for the void fraction calculator.
103 *
104 * @param triangulation The triangulation on which the simulation is being done.
105 * @param input_parameters The parameters for the void fraction calculation.
106 * @param linear_solver_parameters The parameters for the linear solver. This is used to solve the L2 projection of the void fraction
107 * @param particle_handler The particle handler used to manage the particles in the simulation. This object must be provided even if the particles are not used to establish the void fraction.
108 * @param fe_degree The finite element degree used to interpolate the void fraction.
109 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
110 * @param pcout The ConditionalOStream used to print the information.
111 */
112 VoidFractionBase(
113 parallel::DistributedTriangulationBase<dim> *triangulation,
114 std::shared_ptr<Parameters::VoidFractionParameters<dim>> input_parameters,
115 const Parameters::LinearSolver &linear_solver_parameters,
116 Particles::ParticleHandler<dim> *particle_handler,
117 const unsigned int fe_degree,
118 const bool simplex,
119 const ConditionalOStream &pcout)
120 : PhysicsLinearSubequationsSolver(pcout)
121 , dof_handler(*triangulation)
122 , triangulation(triangulation)
123 , void_fraction_parameters(input_parameters)
124 , linear_solver_parameters(linear_solver_parameters)
125 , particle_handler(particle_handler)
126 {
127 if (simplex)
128 {
129 fe = std::make_shared<FE_SimplexP<dim>>(fe_degree);
130 mapping = std::make_shared<MappingFE<dim>>(*fe);
131 quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
132 }
133 else
134 {
135 // Usual case, for quad/hex meshes
136 fe = std::make_shared<FE_Q<dim>>(fe_degree);
137 mapping = std::make_shared<MappingQ<dim>>(fe->degree);
138 if (this->void_fraction_parameters->quadrature_rule ==
139 Parameters::VoidFractionQuadratureRule::gauss)
140 {
141 if (this->void_fraction_parameters->n_quadrature_points == 0)
142 quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
143 else
144 quadrature = std::make_shared<QGauss<dim>>(
145 this->void_fraction_parameters->n_quadrature_points);
146 }
147 if (this->void_fraction_parameters->quadrature_rule ==
148 Parameters::VoidFractionQuadratureRule::gauss_lobatto)
149 {
150 if (this->void_fraction_parameters->n_quadrature_points == 0)
151 quadrature = std::make_shared<QGaussLobatto<dim>>(fe->degree + 2);
152 else if (this->void_fraction_parameters->n_quadrature_points >= 3)
153 quadrature = std::make_shared<QGaussLobatto<dim>>(
154 this->void_fraction_parameters->n_quadrature_points);
155 else
156 throw(std::runtime_error(
157 "For void fraction using Gauss-Lobatto ('gauss-lobatto') quadrature rule, the minimum number of quadrature points is 3"));
158 }
159 }
160 }

268 /**
269 * @brief Calculate the characteristic radius of a sphere with a given volume (area in 2D):
270 * R = (2*dim*V/pi)^(1/dim) / 2
271 *
272 * @param volume value for which the equivalent radius of a sphere is calculated.
273 *
274 * @return Radius of a sphere with equivalent volume

276 static double
277 radius_sphere_equivalent_volume(const double volume)
278 {
279 return 0.5 * pow(2.0 * static_cast<double>(dim) * volume / M_PI,
280 1.0 / static_cast<double>(dim));

284 * @brief Calculate the radius of the QCM averaging sphere
285 *

307 /**
308 * @brief Calculate the intersection measure (volume in 3D and area in 2D) between a particle and an hypersphere.
309 * This is used when calculating the intersection volume between the averaging
310 * sphere and a particle in QCM. The calculation inherently assumes that the
311 * radius of the averaging sphere is larger than that of the particle.
312 *
313 * @param r_particle The radius of the particle
314 * @param r_sphere The radius of the averaging sphere
315 * @param distance_between_spheres The distance between the centers of the spheres
316 *
317 * @return The intersection volume (in 3D) or area (in 2d)
318 */
319
320 static double
321 calculate_intersection_measure(double r_particle,
322 double r_sphere,
323 double distance_between_spheres)
324 {
325 // Particle completely in the reference sphere
326 if (distance_between_spheres <= (r_sphere - r_particle))
327 {
328 if constexpr (dim == 2)
329 return M_PI * Utilities::fixed_power<2>(r_particle);
330 if constexpr (dim == 3)
331 return 4. / 3. * M_PI * Utilities::fixed_power<3>(r_particle);
332 }
333
334 // Particle partially in the reference sphere
335 if ((distance_between_spheres > (r_sphere - r_particle)) &&
336 (distance_between_spheres < (r_sphere + r_particle)))
337 {
338 if constexpr (dim == 2)
339 return particle_circle_intersection_2d(r_particle,
340 r_sphere,
341 distance_between_spheres);
342
343 else if constexpr (dim == 3)
344 return particle_sphere_intersection_3d(r_particle,
345 r_sphere,
346 distance_between_spheres);
347 }
348
349 // Particle completely outside the reference
350 // sphere. The intersection volume is zero.
351 return 0;
352 }

392 /**
393 * @brief Calculate and return the periodic offset distance vector of the domain which is needed
394 * for the periodic boundary conditions using the QCM or SPM for void fraction
395 * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
396 * periodic boundaries. This periodic boundary is then used as the reference
397 * for all particle locations shifted by this offset vector.
398 *
399 * @param[in] boundary_id The ID of one of the periodic boundaries
400 *
401 * @return The periodic offset vector.
402 */
403 inline Tensor<1, dim>
404 get_periodic_offset_distance(unsigned int boundary_id) const
405 {
406 Tensor<1, dim> offset;
407
408 // Iterating over the active cells in the triangulation
409 for (const auto &cell : (*this->triangulation).active_cell_iterators())
410 {
411 if (cell->is_locally_owned() || cell->is_ghost())
412 {
413 if (cell->at_boundary())
414 {
415 // Iterating over cell faces
416 for (unsigned int face_id = 0; face_id < cell->n_faces();
417 ++face_id)
418 {
419 unsigned int face_boundary_id =
420 cell->face(face_id)->boundary_id();
421
422 // Check if face is on the boundary, if so, get
423 // the periodic offset distance for one pair of periodic
424 // faces only since periodic boundaries are aligned with the
425 // direction and only axis are currently allowed
426 if (face_boundary_id == boundary_id)
427 {
428 Point<dim> face_center = cell->face(face_id)->center();
429 auto periodic_cell = cell->periodic_neighbor(face_id);
430 unsigned int periodic_face_id =
431 cell->periodic_neighbor_face_no(face_id);
432 Point<dim> periodic_face_center =
433 periodic_cell->face(periodic_face_id)->center();
434
435 offset = periodic_face_center - face_center;
436
437 return offset;
438 }
439 }
440 }
441 }
442 }
443
444 // A zero tensor is returned in case no cells are found on the periodic
445 // boundaries on this processor. This processor won't handle particle in
446 // cells at periodic boundaries, so it won't affect any computation.
447 return offset;
448 }
