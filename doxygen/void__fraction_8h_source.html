32/**
33 * @brief Calculate the area of intersection between a circular (2D) particle and a circle.
34 *
35 * @param[in] r_particle Radius of the particle
36 *
37 * @param[in] r_circle Radius of the circle
38 *
39 * @param[in] neighbor_distance Distance between the particle and the circle
40 */
41inline double
42particle_circle_intersection_2d(double r_particle,
43 double r_circle,
44 double neighbor_distance)
45{
46 return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
47 cos((pow(neighbor_distance, 2) +
48 pow(r_particle, 2) - pow(r_circle, 2)) /
49 (2 * neighbor_distance * r_particle))) +
50 Utilities::fixed_power<2, double>(r_circle) *
51 Utilities::fixed_power<-1, double>(
52 cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
53 pow(r_circle, 2)) /
54 (2 * neighbor_distance * r_circle))) -
55 0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
56 (neighbor_distance + r_particle - r_circle) *
57 (neighbor_distance - r_particle + r_circle) *
58 (neighbor_distance + r_particle + r_circle));
59}

61/**
62 * @brief Calculate the volume of intersection between a spherical (3D) particle and a sphere.
63 *
64 * @param[in] r_particle Radius of the particle
65 *
66 * @param[in] r_sphere Radius of the sphere
67 *
68 * @param[in] neighbor_distance Distance between the particle and the sphere
69 */
70
71inline double
72particle_sphere_intersection_3d(double r_particle,
73 double r_sphere,
74 double neighbor_distance)
75{
76 return M_PI *
77 Utilities::fixed_power<2, double>(r_sphere + r_particle -
78 neighbor_distance) *
79 (Utilities::fixed_power<2, double>(neighbor_distance) +
80 (2 * neighbor_distance * r_particle) -
81 (3 * Utilities::fixed_power<2, double>(r_particle)) +
82 (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
83 (3 * Utilities::fixed_power<2, double>(r_sphere))) /
84 (12 * neighbor_distance);
85}

87/**
88 * @brief Void fraction calculator. This class manages the calculation of the
89 * void fraction which is used as an auxiliary field for the solvers that solve
90 * the Volume-Averaged Navier-Stokes equations. The present architecture
91 * of the class support all of the different void fraction calculation methods
92 * within a single class instead of building a class hierarchy. This is
93 * because there are only a few void fraction calculation strategies.
94 *
95 * @tparam dim An integer that denotes the number of spatial dimensions.
96 */
97template <int dim>
98class VoidFractionBase : public PhysicsLinearSubequationsSolver
99{
100public:
101 /**
102 * @brief Constructor for the void fraction calculator.
103 *
104 * @param triangulation The triangulation on which the simulation is being done.
105 * @param input_parameters The parameters for the void fraction calculation.
106 * @param linear_solver_parameters The parameters for the linear solver. This is used to solve the L2 projection of the void fraction
107 * @param particle_handler The particle handler used to manage the particles in the simulation. This object must be provided even if the particles are not used to establish the void fraction.
108 * @param fe_degree The finite element degree used to interpolate the void fraction.
109 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
110 * @param pcout The ConditionalOStream used to print the information.
111 */
112 VoidFractionBase(
113 parallel::DistributedTriangulationBase<dim> *triangulation,
114 std::shared_ptr<Parameters::VoidFractionParameters<dim>> input_parameters,
115 const Parameters::LinearSolver &linear_solver_parameters,
116 Particles::ParticleHandler<dim> *particle_handler,
117 const unsigned int fe_degree,
118 const bool simplex,
119 const ConditionalOStream &pcout)
120 : PhysicsLinearSubequationsSolver(pcout)
121 , dof_handler(*triangulation)
122 , triangulation(triangulation)
123 , void_fraction_parameters(input_parameters)
124 , linear_solver_parameters(linear_solver_parameters)
125 , particle_handler(particle_handler)
126 {
127 if (simplex)
128 {
129 fe = std::make_shared<FE_SimplexP<dim>>(fe_degree);
130 mapping = std::make_shared<MappingFE<dim>>(*fe);
131 quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
132 }
133 else
134 {
135 // Usual case, for quad/hex meshes
136 fe = std::make_shared<FE_Q<dim>>(fe_degree);
137 mapping = std::make_shared<MappingQ<dim>>(fe->degree);
138 quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
139 }
140 }

276 /**
277 * @brief Calculate and return the periodic offset distance vector of the domain which is needed
278 * for the periodic boundary conditions using the QCM or SPM for void fraction
279 * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
280 * periodic boundaries. This periodic boundary is then used as the reference
281 * for all particle locations shifted by this offset vector.
282 *
283 * @param[in] boundary_id The ID of one of the periodic boundaries
284 *
285 * @return The periodic offset vector.
286 */
287 inline Tensor<1, dim>
288 get_periodic_offset_distance(unsigned int boundary_id) const
289 {
290 Tensor<1, dim> offset;
291
292 // Iterating over the active cells in the triangulation
293 for (const auto &cell : (*this->triangulation).active_cell_iterators())
294 {
295 if (cell->is_locally_owned() || cell->is_ghost())
296 {
297 if (cell->at_boundary())
298 {
299 // Iterating over cell faces
300 for (unsigned int face_id = 0; face_id < cell->n_faces();
301 ++face_id)
302 {
303 unsigned int face_boundary_id =
304 cell->face(face_id)->boundary_id();
305
306 // Check if face is on the boundary, if so, get
307 // the periodic offset distance for one pair of periodic
308 // faces only since periodic boundaries are aligned with the
309 // direction and only axis are currently allowed
310 if (face_boundary_id == boundary_id)
311 {
312 Point<dim> face_center = cell->face(face_id)->center();
313 auto periodic_cell = cell->periodic_neighbor(face_id);
314 unsigned int periodic_face_id =
315 cell->periodic_neighbor_face_no(face_id);
316 Point<dim> periodic_face_center =
317 periodic_cell->face(periodic_face_id)->center();
318
319 offset = periodic_face_center - face_center;
320
321 return offset;
322 }
323 }
324 }
325 }
326 }
327
328 // A zero tensor is returned in case no cells are found on the periodic
329 // boundaries on this processor. This processor won't handle particle in
330 // cells at periodic boundaries, so it won't affect any computation.
331 return offset;
332 }
