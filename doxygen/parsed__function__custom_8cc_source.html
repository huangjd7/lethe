20template <int n_components>
21void
22ParsedFunctionCustom<n_components>::declare_parameters(ParameterHandler &prm)
23{
24 Assert(n_components > 0, ExcZero());
25
26 // The expression of the variables
27 vnames = "c_";
28 for (unsigned int i = 1; i < n_components; i++)
29 vnames = vnames + ",c_" + std::to_string(i);
30 std::cout << vnames << std::endl;
31 prm.declare_entry(
32 "Variable names",
33 vnames,
34 Patterns::Anything(),
35 "The names of the variables as they will be used in the "
36 "function, separated by commas. By default, the names of variables "
37 "with which the function will be evaluated are the `c_i' (i from 0 to "
38 "n_components-1) and `t' for time. You can then use these variable names "
39 "in your function expression and they will be "
40 "replaced by the values of these variables at which the function is "
41 "currently evaluated. However, you can also choose a different set "
42 "of names for the variables at which to evaluate your function "
43 "expression.");
44
45 // The expression of the function
46 std::string expr = "0";
47 for (unsigned int i = 1; i < n_components; ++i)
48 expr += "; 0";
49 prm.declare_entry(
50 "Function expression",
51 expr,
52 Patterns::Anything(),
53 "The formula that denotes the function you want to evaluate for "
54 "particular values of the variables. This expression "
55 "may contain any of the usual operations such as addition or "
56 "multiplication, as well as all of the common functions such as "
57 "`sin' or `cos'. In addition, it may contain expressions like "
58 "`if(x>0, 1, -1)' where the expression evaluates to the second "
59 "argument if the first argument is true, and to the third argument "
60 "otherwise. "
61 "If the function you are describing represents a vector-valued "
62 "function with multiple components, then separate the expressions "
63 "for individual components by a semicolon.");
64
65 prm.declare_entry(
66 "Function constants",
67 "",
68 Patterns::Anything(),
69 "Sometimes it is convenient to use symbolic constants in the "
70 "expression that describes the function, rather than having to "
71 "use its numeric value everywhere the constant appears. These "
72 "values can be defined using this parameter, in the form "
73 "`var1=value1, var2=value2, ...'."
74 "\n\n"
75 "A typical example would be to set this runtime parameter to "

77 "actual formula. (That said, for convenience this class actually "

82template <int n_components>
83void
84ParsedFunctionCustom<n_components>::initialize()
85{
86 if (!initialized)
87 throw ExcMessage(
88 "The required fields for muParser are not initialized yet. "
89 "This function should be called only after that is done.");
90 std::vector<std::string> const_list =
91 Utilities::split_string_list(constants_list, ',');
92 std::map<std::string, double> constants;
93 for (const auto &constant : const_list)
94 {
95 std::vector<std::string> this_c =
96 Utilities::split_string_list(constant, '=');
97 AssertThrow(this_c.size() == 2,
98 ExcMessage("The list of constants, <" + constants_list +
99 ">, is not a comma-separated list of "
100 "entries of the form 'name=value'."));
101 constants[this_c[0]] = Utilities::string_to_double(this_c[1]);
102 }
103 // set pi and Pi as synonyms for the corresponding value. note that
104 // this overrides any value a user may have given
105 constants["pi"] = numbers::PI;
106 constants["Pi"] = numbers::PI;
107
108 const unsigned int nn = (Utilities::split_string_list(vnames)).size();
109 switch (nn)
110 {
111 case n_components:
112 // Time independent function
113 this->::internalFunctionParserCustom::ParserImplementation<
114 n_components>::initialize(vnames,
115 Utilities::split_string_list(expression,
116 ';'),
117 constants);
118 break;
119 case n_components + 1:
120 // Time dependent function
121 this->::internalFunctionParserCustom::ParserImplementation<
122 n_components>::initialize(vnames,
123 Utilities::split_string_list(expression,
124 ';'),
125 constants,
126 true);
127 break;
128 default:
129 AssertThrow(false,
130 ExcMessage(
131 "The list of variables specified is <" + vnames +
132 "> which is a list of length " +
133 Utilities::int_to_string(nn) +
134 " but it has to be a list of length equal to" +
135 " either n_components (for a time-independent function)" +
136 " or n_components+1 (for a time-dependent function)."));
137 }
138}

183template <int n_components>
184Tensor<1, n_components>
185ParsedFunctionCustom<n_components>::gradient(const Tensor<1, n_components> &p,
186 unsigned int comp) const
187{
188 Tensor<1, n_components> grad;
189 // FourthOrder:
190 Tensor<1, n_components> q1, q2, q3, q4;
191 for (unsigned int i = 0; i < n_components; ++i)
192 {
193 q2 = p + ht[i];
194 q1 = q2 + ht[i];
195 q3 = p - ht[i];
196 q4 = q3 - ht[i];
197 grad[i] = (-this->value(q1, comp) + 8 * this->value(q2, comp) -
198 8 * this->value(q3, comp) + this->value(q4, comp)) /
199 (12 * h);
200 }
201 return grad;
202}

204template <int n_components>
205void
206ParsedFunctionCustom<n_components>::vector_gradient(
207 const Tensor<1, n_components> &p,
208 Tensor<2, n_components> &gradients) const
209{
210 Point<n_components> q1, q2, q3, q4;
211 Tensor<1, n_components> v1{}, v2{}, v3{}, v4{};
212 const double h_inv_12 = 1. / (12 * h);
213 for (unsigned int i = 0; i < n_components; ++i)
214 {
215 q2 = p + ht[i];
216 q1 = q2 + ht[i];
217 q3 = p - ht[i];
218 q4 = q3 - ht[i];
219 this->vector_value(q1, v1);
220 this->vector_value(q2, v2);
221 this->vector_value(q3, v3);
222 this->vector_value(q4, v4);
223
224 for (unsigned int comp = 0; comp < n_components; ++comp)
225 gradients[comp][i] =
226 (-v1[comp] + 8 * v2[comp] - 8 * v3[comp] + v4[comp]) * h_inv_12;
227 }
228}
