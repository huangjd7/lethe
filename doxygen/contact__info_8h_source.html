14/**
15 * @brief Handle the information related to the calculation of the
16 * particle-particle contact force. Notably it is responsible for storing
17 * information that has to be preserved over multiple iterations of a contact,
18 * namely everything related to tangential overlaps
19 */
20template <int dim>
21struct particle_particle_contact_info
22{
23 Particles::ParticleIterator<dim> particle_one;
24 Particles::ParticleIterator<dim> particle_two;
25 Tensor<1, 3> tangential_overlap;
26 Tensor<1, 3> rolling_resistance_spring_torque;
27};

29template <int dim>
30class particle_wall_contact_info
31{
32public:
33 /**
34 * @brief Construct the particle_wall_contact_info using the particle
35 * iterator, the normal vector of the wall, a point on that boundary and the
36 * boundary id. This is the commonly used constructor since it houses all the
37 * information required to perform the contact calculation.
38 *
39 * @param particle The iterator to the particle in contact with the wall.
40 * @param normal_vector The outward pointing normal vector on the wall.
41 * @param point_on_boundary A point that lies on the face.
42 * @param boundary_id The boundary id. This id corresponds to the number
43 * attributed to the boundary condition.
44 *
45 * TODO: This should be a struct and normal_overlap, normal_relative_velocity
46 * and tangential_relative_velocity should be removed and be calculated on the
47 * fly as done for pp forces with particle_particle_contact_info
48 */
49 particle_wall_contact_info(const Particles::ParticleIterator<dim> &particle,
50 const Tensor<1, 3> &normal_vector,
51 const Point<3> &point_on_boundary,
52 const types::boundary_id &boundary_id)
53 : particle(particle)
54 , normal_vector(normal_vector)
55 , point_on_boundary(point_on_boundary)
56 , boundary_id(boundary_id)
57 , normal_overlap(0)

59 , tangential_overlap({0, 0, 0})
60 , tangential_relative_velocity({0, 0, 0})
