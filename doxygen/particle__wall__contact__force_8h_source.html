113 /**
114 * @brief This function is used to find the projection of vector_a on
115 * vector_b
116 * @param vector_a A vector which is going to be projected on vector_b
117 * @param vector_b The projection vector of vector_a
118 * @return The projection of vector_a on vector_b
119 */
120 inline Tensor<1, 3>
121 find_projection(const Tensor<1, 3> &vector_a, const Tensor<1, 3> &vector_b)
122 {
123 Tensor<1, 3> vector_c;
124 vector_c = ((vector_a * vector_b) / (vector_b.norm_square())) * vector_b;
125
126 return vector_c;
127 }

168 /**
169 * @brief Apply the calculated force and torque on the particle in
170 * contact with the given wall, for both non-linear and linear contact force
171 * calculations
172 *
173 * @param forces_and_torques A tuple which contains: 1, normal force, 2,
174 * tangential force, 3, tangential torque and 4, rolling resistance torque of
175 * a contact pair
176 * @param particle_torque Torque acting on particle
177 * @param particle_force Force acting on particle
178 * @param point_on_boundary Contact point on the wall
179 * @param boundary_id ID of the boundary
180 */
181 inline void
182 apply_force_and_torque(
183 const std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
184 &forces_and_torques,
185 Tensor<1, 3> &particle_torque,
186 Tensor<1, 3> &particle_force,
187 const Point<3> &point_on_boundary,
188 int boundary_id = 0)
189 {
190 // Getting the values from the forces_and_torques tuple, which are: 1,
191 // normal force, 2, tangential force, 3, tangential torque and 4, rolling
192 // resistance torque
193 Tensor<1, 3> normal_force = std::get<0>(forces_and_torques);
194 Tensor<1, 3> tangential_force = std::get<1>(forces_and_torques);
195 Tensor<1, 3> tangential_torque = std::get<2>(forces_and_torques);
196 Tensor<1, 3> rolling_resistance_torque = std::get<3>(forces_and_torques);
197
198 // Calculation of total force
199 Tensor<1, 3> total_force = normal_force + tangential_force;
200
201 calculate_force_and_torque_on_boundary(boundary_id,
202 total_force,
203 point_on_boundary);
204
205 // Updating the force of particles in the particle handler
206 // Since the forces were calculated on the wall, we use the -= operator
207 particle_force -= total_force;
208
209 // Updating the torque acting on particles
210 // The torque was direcly calculated on the particle, thus we use the +=
211 // operator
212 particle_torque += tangential_torque + rolling_resistance_torque;
213 }
