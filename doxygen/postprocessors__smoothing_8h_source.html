36/**
37 * A base class that holds the dof_handler, fe_values and the simulation output
38 * to calculate post-processed parameters
39 *
40 * @tparam dim An integer that denotes the dimension of the space in which
41 * the variable being smoothed is solved
42 * @param triangulation Flow triangulation
43 * @param simulation_parameters The simulation parameters
44 * @param number_quadrature_points The number of quadrature points
45 * @param mpi_communicator Allowing communication between cores
46 */
47
48template <int dim, typename VectorType>
49class PostProcessorSmoothing
50{
51public:
52 // Member functions
53 PostProcessorSmoothing(
54 const parallel::DistributedTriangulationBase<dim> &triangulation,
55 const SimulationParameters<dim> &simulation_parameters,
56 const unsigned int &number_quadrature_points);
57
58 /**
59 * @brief Generates the mass matrix, that is independent of the physics.
60 */
61 void
62 generate_mass_matrix();
63
64 /**
65 * @brief Generates the right hand side based on the fluid's solution.
66 */
67 virtual void
68 generate_rhs(const VectorType &,

70 std::shared_ptr<Mapping<dim>>)

72
73 /**

76 const GlobalVectorType &

108};
